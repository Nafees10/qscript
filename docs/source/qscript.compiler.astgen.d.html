16 struct ASTGen{
17 	/// constructor
18 	///

21 	/// DataTypes of arguments it's being called with.
22 	/// 
23 	/// `onArgsTypeOkFunction` is a pointer to a function that will return true if the argument types for a pre-defined function
24 	/// are ok, else, false. It receives the functionName, and the argument types
25 	this (DataType delegate (string, DataType[]) onGetReturnTypeFunction,
26 		bool delegate(string, DataType[]) onArgsTypeOkFunction){
27 		onArgsTypeOk = onArgsTypeOkFunction;
28 		onGetFunctionReturnType = onGetReturnTypeFunction;

35 	/// generates an AST representing a script.
36 	/// 
37 	/// `tokens` is the TokenList to generate AST from
38 	/// `errors` is the array in which errors will be put
39 	/// 
40 	/// The script must be converted to tokens using `qscript.compiler.tokengen.toTokens`
41 	/// If any errors occur, they will be contanied in `qscript.compiler.misc.`
42 	public ScriptNode generateScriptAST(TokenList scriptTokens, ref CompileError[] errors){
43 		tokens = scriptTokens;
44 		ScriptNode scriptNode;
45 		LinkedList!FunctionNode functions = new LinkedList!FunctionNode;
46 		// scan for return types & arg types of script-defined functions
47 		scanTokensForFunctionReturnArgTypes();
48 		// go through the script, compile function nodes, link them to this node
49 		index = 0;
50 		for (uinteger lastIndex = tokens.tokens.length - 1; index < tokens.tokens.length; index ++){
51 			// look for TokenType.Keyword where token.token == "function"
52 			if (tokens.tokens[index].type == Token.Type.Keyword && tokens.tokens[index].token == "function"){
53 				uinteger errorCount = compileErrors.count;
54 				FunctionNode functionNode = generateFunctionAST();
55 				index --;
56 				// check if error free
57 				if (compileErrors.count == errorCount){
58 					functions.append(functionNode);
59 				}
60 			}
61 		}
62 		scriptNode = ScriptNode(functions.toArray);
63 		.destroy(functions);
64 		errors = compileErrors.toArray;
65 		.destroy(compileErrors);
66 		return scriptNode;
67 	}

83 		struct{
84 			/// stores data types for variable in currently-being-converted function
85 			private DataType[string] varDataTypes;
86 			/// stores the scope depth count for each var
87 			private uinteger[string] varScopeDepth;
88 			/// stores current scope
89 			private uinteger scopeCount = 0;
90 			/// adds a var, throws Exception if var with same name already exists
91 			void addVarType(string name, DataType type){
92 				if (name in varDataTypes){
93 					throw new Exception("variable '"~name~"' declared twice");
94 				}
95 				varDataTypes[name] = type;
96 				varScopeDepth[name] = scopeCount;
97 			}
98 			/// returns data type of a var, throws Exception if var doesnt exist
99 			DataType getVarType(string name){
100 				if (name in varDataTypes){
101 					return varDataTypes[name];
102 				}else{
103 					throw new Exception("variable '"~name~"' not declared");
104 				}
105 			}
106 			/// removes vars of last scope, and decreases scopeCount
107 			void removeLastScope(){
108 				foreach (key; varScopeDepth.keys){
109 					if (varScopeDepth[key] == scopeCount){
110 						varScopeDepth.remove(key);
111 						// remove from varDataTypes too!
112 						varDataTypes.remove(key);
113 					}
114 				}
115 				if (scopeCount > 0){
116 					scopeCount --;
117 				}
118 			}
119 			/// increases scope count
120 			void increaseScopeCount(){
121 				scopeCount ++;
122 			}

124 		/// scans tokens for script-defined functions' return types and argument types
125 		/// 
126 		/// keep in mind that it messes up the value of `index`! so only call it before anything else, and reset `index` then.
127 		void scanTokensForFunctionReturnArgTypes(){
128 			LinkedList!DataType argTypes = new LinkedList!DataType;
129 			for (index = 0; index < tokens.tokens.length; index ++){
130 				if (tokens.tokens[index].type == Token.Type.Keyword && tokens.tokens[index].token == "function"){
131 					index ++;
132 					DataType returnType;
133 					try{
134 						returnType = readType();
135 					}catch(Exception e){
136 						compileErrors.append(CompileError(tokens.getTokenLine(index), e.msg));
137 						.destroy (e);
138 					}
139 					string fName = tokens.tokens[index].token;
140 					functionReturnTypes[fName] = returnType;
141 					// now for the args
142 					index ++; // skip the name
143 					if (tokens.tokens[index].type == Token.Type.ParanthesesOpen){
144 						// it's got args
145 						uinteger brackEnd = tokens.tokens.bracketPos(index);
146 						for (index += 1; index < brackEnd; index ++){
147 							DataType argType;
148 							try{
149 								argType = readType();
150 							}catch (Exception e){
151 								compileErrors.append(CompileError(tokens.getTokenLine(index), e.msg));
152 								.destroy (e);
153 							}
154 							argTypes.append(argType);
155 							// now should be the arg_name, ignore that
156 							assert (tokens.tokens[index].type == Token.Type.Identifier,
157 								"argument type should be followed by argument name");
158 							index ++;
159 							// now skip the comma, or could be a parenthesesEnd
160 							assert (tokens.tokens[index].type == Token.Type.ParanthesesClose ||
161 								tokens.tokens[index].type == Token.Type.Comma,
162 								"arguments must be separated using a comma");
163 							// index ++ in for statement does the job this time
164 						}
165 						functionArgTypes[fName] = argTypes.toArray;
166 						argTypes.clear;
167 					}else{
168 						// no args
169 						functionArgTypes[fName] = [];
170 					}
171 				}
172 				// skip brackets
173 				if ([Token.Type.BlockStart, Token.Type.IndexBracketOpen, Token.Type.ParanthesesOpen].
174 					hasElement(tokens.tokens[index].type)){
175 					index = tokens.tokens.bracketPos(index);
176 				}
177 			}
178 			.destroy (argTypes);
179 		}
180 		/// returns return type of a function, if function doesnt exist, throws Exception
181 		DataType getFunctionReturnType(string functionName, DataType[] argTypes){
182 			/// stores whether the all the tokens have been scanned for function return types
183 			static bool scannedAllTokens = false;
184 			if (functionName in functionReturnTypes){
185 				return functionReturnTypes[functionName];
186 			}else{
187 				if (onGetFunctionReturnType !is null){
188 					try{
189 						return onGetFunctionReturnType(functionName, argTypes.dup);
190 					}catch (Exception e){
191 						compileErrors.append (CompileError(tokens.getTokenLine(index), e.msg));
192 						return DataType(DataType.Type.Void);
193 					}
194 
195 				}else{
196 					throw new Exception("function '"~functionName~"' not defined");
197 				}
198 			}
199 		}
200 		/// matches arguments' types to see if they can be used for a function
201 		/// 
202 		/// in case they dont match, appends error to compileErrors
203 		bool matchFunctionArgTypes(string functionName, DataType[] argTypes){
204 			bool r = true;
205 			// check if is predefined
206 			if (functionName in functionArgTypes){
207 				DataType[] acceptableTypes = functionArgTypes[functionName];
208 				if (argTypes.length != acceptableTypes.length){
209 					compileErrors.append(
210 						CompileError(tokens.getTokenLine(index), "function '"~functionName~"' called with invalid number of arguments"));
211 					r = false;
212 				}else{
213 					if (!matchArguments(acceptableTypes.dup, argTypes.dup)){
214 						compileErrors.append(CompileError(tokens.getTokenLine(index), "function '"~functionName~"' called with invalid arguments"));
215 						return false;
216 					}else{
217 						return true;
218 					}
219 				}
220 			}else{
221 				// use onArgsTypeOk
222 				if (onArgsTypeOk !is null){
223 					try{
224 						if (!onArgsTypeOk(functionName, argTypes)){
225 							compileErrors.append(CompileError(tokens.getTokenLine(index),
226 									"function '"~functionName~"' called with invalid arguments"));
227 							return false;
228 						}
229 					}catch (Exception e){
230 						compileErrors.append(CompileError(tokens.getTokenLine(index), e.msg));
231 						return false;
232 					}
233 
234 				}else{
235 					compileErrors.append(CompileError(tokens.getTokenLine(index), "function '"~functionName~"' not defined"));
236 				}
237 			}
238 			return r;
239 		}
240 		/// reads a type from TokensList
241 		/// 
242 		/// returns type in DataType struct, changes `index` to token after last token of type
243 		DataType readType(){
244 			uinteger startIndex = index;
245 			// the first token has to be the type (int, string, double)
246 			index ++;
247 			// then comes the brackets making it an array
248 			if (tokens.tokens[index].type == Token.Type.IndexBracketOpen){
249 				for (; index < tokens.tokens.length; index ++){
250 					if (tokens.tokens[index].type != Token.Type.IndexBracketOpen &&
251 						tokens.tokens[index].type != Token.Type.IndexBracketClose){
252 						break;
253 					}
254 				}
255 			}
256 			string sType = tokens.toString(tokens.tokens[startIndex .. index]);
257 			return DataType(sType);
258 		}
259 		/// generates AST for a function definition 
260 		/// 
261 		/// changes `index` to the token after function definition
262 		FunctionNode generateFunctionAST(){
263 			FunctionNode functionNode;
264 			// make sure it's a function
265 			increaseScopeCount();
266 			if (tokens.tokens[index].type == Token.Type.Keyword && tokens.tokens[index].token == "function"){
267 				// read the type
268 				index++;
269 				try{
270 					functionNode.returnType = readType();
271 				}catch(Exception e){
272 					compileErrors.append(CompileError(tokens.getTokenLine(index), e.msg));
273 					.destroy (e);
274 				}
275 				// above functionCall moves index to fName
276 				// now index is at function name
277 				functionNode.name = tokens.tokens[index].token;
278 				index ++;
279 				// now for the argument types
280 				if (tokens.tokens[index].type == Token.Type.ParanthesesOpen){
281 					LinkedList!(FunctionNode.Argument) argList = new LinkedList!(FunctionNode.Argument);
282 					uinteger brackEnd = tokens.tokens.bracketPos(index);
283 					bool commaExpected = false;
284 					for (index ++; index < brackEnd; index ++){
285 						if (commaExpected){
286 							if (tokens.tokens[index].type == Token.Type.Comma){
287 								commaExpected = false;
288 								continue;
289 							}else{
290 								compileErrors.append(CompileError(tokens.getTokenLine(index),
291 										"arguments must be separated using a comma"));
292 							}
293 						}else{
294 							// data type + arg_name expected
295 							// read the type
296 							FunctionNode.Argument arg;
297 							try{
298 								arg.argType = readType();
299 							}catch(Exception e){
300 								compileErrors.append(CompileError(tokens.getTokenLine(index), e.msg));
301 								.destroy (e);
302 							}
303 							// the type must not be void, script-defined functions must all be static typed
304 							if (arg.argType.type == DataType.Type.Void){
305 								compileErrors.append(CompileError(tokens.getTokenLine(index),
306 										"script-defined functions can not receive arguments of type void"));
307 							}
308 							// now the arg_name
309 							arg.argName = tokens.tokens[index].token;
310 							commaExpected = true;
311 							// add it to list
312 							argList.append (arg);
313 							addVarType(arg.argName, arg.argType);
314 						}
315 					}
316 					// put args in list
317 					functionNode.arguments = argList.toArray;
318 					.destroy(argList);
319 					index = brackEnd+1;
320 				}
321 				if (tokens.tokens[index].type == Token.Type.BlockStart){
322 					// no args, just body
323 					functionNode.bodyBlock = generateBlockAST();
324 				}else{
325 					compileErrors.append(CompileError(tokens.getTokenLine(index), "function has no body"));
326 				}
327 			}else{
328 				compileErrors.append(CompileError(tokens.getTokenLine(index), "not a function definition"));
329 				index ++;
330 			}
331 			removeLastScope();
332 			return functionNode;
333 		}

335 		/// generates AST for a {block-of-code}
336 		/// 
337 		/// changes `index` to token after block end
338 		BlockNode generateBlockAST(){
339 			BlockNode blockNode;
340 			// make sure it's a block
341 			if (tokens.tokens[index].type == Token.Type.BlockStart){
342 				increaseScopeCount();
343 				uinteger brackEnd = tokens.tokens.bracketPos!(true)(index);
344 				LinkedList!StatementNode statements = new LinkedList!StatementNode;
345 				// read statements
346 				index++;
347 				uinteger errorCount = compileErrors.count;
348 				while (index < brackEnd){
349 					statements.append(generateStatementAST());
350 					// check if error added
351 					if (errorCount != compileErrors.count){
352 						break;
353 					}
354 				}
355 				// put them all in block
356 				blockNode.statements = statements.toArray;
357 				.destroy(statements);
358 				index = brackEnd+1;
359 
360 				removeLastScope();
361 			}else{
362 				compileErrors.append(CompileError(tokens.getTokenLine(index), "not a block"));
363 				index ++;
364 			}
365 			return blockNode;
366 		}

368 		/// generates AST for one statement
369 		StatementNode generateStatementAST(){
370 			// identify the type, call the appropriate function
371 			if (tokens.tokens[index].type == Token.Type.BlockStart){
372 				// block
373 				return StatementNode(generateBlockAST());
374 			}else if (tokens.tokens[index].type == Token.Type.Keyword){
375 				if (tokens.tokens[index].token == "if"){
376 					// if statement
377 					return StatementNode(generateIfAST());
378 				}else if (tokens.tokens[index].token == "while"){
379 					// while statement
380 					return StatementNode(generateWhileAST());
381 				}
382 			}else if (tokens.tokens[index].type == Token.Type.DataType){
383 				// var declare
384 				return StatementNode(generateVarDeclareAST());
385 			}else if (tokens.tokens[index].type == Token.Type.Identifier &&
386 				tokens.tokens[index+1].type == Token.Type.ParanthesesOpen){
387 				// is a function call
388 				return StatementNode(generateFunctionCallAST());
389 			}else if (tokens.tokens[index].type == Token.Type.Identifier){
390 				// assignment
391 				return StatementNode(generateAssignmentAST());
392 			}
393 			compileErrors.append (CompileError(tokens.getTokenLine(index), "invalid statement"));
394 			index ++;
395 			return StatementNode();
396 		}

398 		/// generates AST for function call, changes `index` to token after statementEnd
399 		FunctionCallNode generateFunctionCallAST(){
400 			FunctionCallNode functionCallNode = FunctionCallNode();
401 			// check if is function call
402 			if (tokens.tokens[index].type == Token.Type.Identifier &&
403 				tokens.tokens[index + 1].type == Token.Type.ParanthesesOpen){
404 				uinteger brackEnd = tokens.tokens.bracketPos(index + 1);
405 				functionCallNode.fName = tokens.tokens[index].token;
406 				// now for the arguments
407 				index+=2;
408 				if (tokens.tokens[index].type == Token.Type.ParanthesesClose){
409 					// has no args
410 					functionCallNode.arguments = [];
411 				}else{
412 					LinkedList!CodeNode args = new LinkedList!CodeNode;
413 					for (; ; index ++){
414 						args.append(generateCodeAST());
415 						if (tokens.tokens[index].type == Token.Type.ParanthesesClose){
416 							break;
417 						}
418 					}
419 					functionCallNode.arguments = args.toArray;
420 					.destroy(args);
421 				}
422 				// match argument types
423 				DataType[] argTypes;
424 				argTypes.length = functionCallNode.arguments.length;
425 				foreach(i, arg; functionCallNode.arguments){
426 					argTypes[i] = arg.returnType;
427 				}
428 				matchFunctionArgTypes(functionCallNode.fName, argTypes);
429 				// then the return type
430 				try{
431 					functionCallNode.returnType = getFunctionReturnType(functionCallNode.fName, argTypes);
432 				}catch (Exception e){
433 					compileErrors.append(CompileError(tokens.getTokenLine(index), e.msg));
434 					.destroy(e);
435 				}
436 				// move index to bracketend or semicolon
437 				index = brackEnd+1;
438 				if (tokens.tokens[index].type == Token.Type.StatementEnd){
439 					index ++;
440 				}
441 				return functionCallNode;
442 			}else{
443 				compileErrors.append(CompileError(tokens.getTokenLine(index), "not a valid function call"));
444 				index ++;
445 			}
446 			return functionCallNode;
447 		}

449 		/// generates AST for "actual code" like `2 + 2 - 6`.
450 		/// 
451 		/// terminates reading when on a Token that is comma, paranthesesEnd, semicolon, index/block bracket close
452 		CodeNode generateCodeAST(){
453 			CodeNode lastNode = null;
454 			bool separatorExpected = false;
455 			for ( ; ; index ++){
456 				Token token = tokens.tokens[index];
457 				// check if has to terminate
458 				if ([Token.Type.Comma, Token.Type.IndexBracketClose, Token.Type.BlockEnd, Token.Type.ParanthesesClose,
459 					Token.Type.StatementEnd].hasElement(token.type)){
460 					break;
461 				}
462 				if (!separatorExpected){
463 					// an identifier, or literal (i.e some data) was expected
464 					lastNode = generateNodeAST();
465 					index --;
466 					separatorExpected = true;
467 				}else{
468 					// an operator or something that deals with data was expected.
469 					// check if there's an [...] to read the array
470 					if (token.type == Token.Type.IndexBracketOpen){
471 						// read the index
472 						uinteger brackEnd = tokens.tokens.bracketPos(index);
473 						index ++;
474 						CodeNode indexNode = generateCodeAST();
475 						// make sure it's a read-element, not a make-array
476 						if (index != brackEnd){
477 							compileErrors.append (CompileError(tokens.getTokenLine(index), "unexpected tokens"));
478 							index++;
479 							break;
480 						}
481 						// make sure lastNode is an array with enough dimensions
482 						if (lastNode.returnType.arrayNestCount == 0 && lastNode.returnType.type != DataType.Type.String){
483 							compileErrors.append (CompileError(tokens.getTokenLine(index),
484 									"can not read that many dimensions from that array"));
485 						}
486 						lastNode = CodeNode(ReadElement(lastNode, indexNode));
487 						continue;
488 					}
489 					separatorExpected = false;
490 					if (token.type == Token.Type.Operator){
491 						lastNode = CodeNode(generateOperatorAST(lastNode));
492 						index --;
493 						// in case of operators, generateOperatorAST takes the 2nd operand (!separator), itself, so the next
494 						// token, if any, is to be a separator
495 						separatorExpected = true;
496 						continue;
497 					}else{
498 						break;
499 					}
500 				}
501 			}
502 			return lastNode;
503 		}

505 		/// generates AST for operators like +, -...
506 		/// 
507 		/// `firstOperand` is the first operand for the operator.
508 		/// `index` is the index of the token which is the operator
509 		/// 
510 		/// changes `index` to the index of the token after the last token related to the operator
511 		OperatorNode generateOperatorAST(CodeNode firstOperand){
512 			OperatorNode operator;
513 			// make sure it's an operator, and there is a second operand
514 			if (tokens.tokens[index].type == Token.Type.Operator && index+1 < tokens.tokens.length){
515 				// read the next operand
516 				CodeNode secondOperand;
517 				string operatorString = tokens.tokens[index].token;
518 				index ++;
519 				secondOperand = generateNodeAST();
520 				// put both operands under one node
521 				operator = OperatorNode(operatorString, firstOperand, secondOperand);
522 			}else{
523 				compileErrors.append(CompileError(tokens.getTokenLine(index), "no second operand found"));
524 				index ++;
525 			}
526 			return operator;
527 		}

550 		/// generates AST for assignment operator
551 		AssignmentNode generateAssignmentAST(){
552 			AssignmentNode assignment;
553 			// get the variable to assign to
554 			CodeNode varCodeNode = generateCodeAST();
555 			VariableNode var;
556 			// make sure it's a var
557 			CodeNode[] indexes;
558 			if (varCodeNode.type == CodeNode.Type.ReadElement){
559 				LinkedList!CodeNode indexesList = new LinkedList!CodeNode;
560 				ReadElement indexRead;
561 				while (varCodeNode.type == CodeNode.Type.ReadElement){
562 					indexRead = varCodeNode.node!(CodeNode.Type.ReadElement);
563 					indexesList.append(indexRead.index);
564 					varCodeNode = indexRead.readFromNode;
565 				}
566 				indexes = indexesList.toArray.reverseArray;
567 			}
568 			var = varCodeNode.node!(CodeNode.Type.Variable);
569 			// now at index, the token should be a `=` operator
570 			if (tokens.tokens[index].type == Token.Type.AssignmentOperator){
571 				// everything's ok till the `=` operator
572 				index++;
573 				CodeNode val = generateCodeAST();
574 				assignment = AssignmentNode(var, indexes, val);
575 				// make sure it's followed by a semicolon
576 				if (tokens.tokens[index].type != Token.Type.StatementEnd){
577 					compileErrors.append(CompileError(tokens.getTokenLine(index),
578 							"assingment statement not followed by semicolon"));
579 				}else{
580 					// skip the semicolon too
581 					index++;
582 				}
583 				// make sure the variable data type matches the val's data type
584 				DataType varType;
585 				try{
586 					varType = getVarType(var.varName);
587 				}catch (Exception e){
588 					compileErrors.append(CompileError(tokens.getTokenLine(index-1), e.msg));
589 				}
590 				// also need to check if the var was declared a array or not, and if it's being set a correct type or not
591 				if (varType.arrayNestCount < indexes.length){
592 					compileErrors.append(CompileError(tokens.getTokenLine(index-1),
593 							"variable dimensions in assignment differs from declaration"));
594 				}else{
595 					// continue with checking
596 					DataType expectedType = DataType(varType.type, varType.arrayNestCount - indexes.length);
597 					// make sure it matches
598 					if (expectedType != val.returnType){
599 						compileErrors.append(CompileError(tokens.getTokenLine(index-1),
600 								"cannot assign value with different data type to variable"));
601 					}
602 				}
603 			}else{
604 				compileErrors.append(CompileError(tokens.getTokenLine(index), "not an assignment statement"));
605 				index ++;
606 			}
607 			return assignment;
608 		}

611 		VarDeclareNode generateVarDeclareAST(){
612 			VarDeclareNode varDeclare;
613 			// make sure it's a var declaration, and the vars are enclosed in parantheses
614 			if (DATA_TYPES.hasElement(tokens.tokens[index].token)){
615 				// read the type
616 				DataType type;
617 				try{
618 					type = readType();
619 				}catch(Exception e){
620 					compileErrors.append(CompileError(tokens.getTokenLine(index), e.msg));
621 					.destroy (e);
622 				}
623 				// make sure next token = bracket
624 				if (tokens.tokens[index].type == Token.Type.ParanthesesOpen){
625 					// now go through all vars, check if they are vars, and are aeparated by comma
626 					uinteger brackEnd = tokens.tokens.bracketPos!true(index);
627 					bool commaExpected = false;
628 					LinkedList!string vars = new LinkedList!string;
629 					for (index = index+1; index < brackEnd; index ++){
630 						Token token = tokens.tokens[index];
631 						if (commaExpected){
632 							if (token.type != Token.Type.Comma){
633 								compileErrors.append(CompileError(tokens.getTokenLine(index),
634 										"variable names in declaration must be separated by a comma"));
635 							}
636 							commaExpected = false;
637 						}else{
638 							if (token.type != Token.Type.Identifier){
639 								compileErrors.append(CompileError(tokens.getTokenLine(index),
640 										"variable name expected in varaible declaration, unexpected token found"));
641 							}else{
642 								vars.append(token.token);
643 								try{
644 									addVarType(token.token, type);
645 								}catch (Exception e){
646 									compileErrors.append(CompileError(tokens.getTokenLine(index), e.msg));
647 									.destroy (e);
648 								}
649 							}
650 							commaExpected = true;
651 						}
652 					}
653 					varDeclare = VarDeclareNode(type, vars.toArray);
654 					.destroy(vars);
655 					// skip the semicolon
656 					if (tokens.tokens[brackEnd+1].type == Token.Type.StatementEnd){
657 						index = brackEnd+2;
658 					}else{
659 						compileErrors.append(CompileError(tokens.getTokenLine(brackEnd+1),
660 									"variable declaration not followed by semicolon"));
661 					}
662 				}else{
663 					compileErrors.append(CompileError(tokens.getTokenLine(index), "invalid variable declaration"));
664 				}
665 			}else{
666 				compileErrors.append(CompileError(tokens.getTokenLine(index), "invalid variable declaration"));
667 				index ++;
668 			}
669 			return varDeclare;

674 			IfNode ifNode;
675 			// check if is an if
676 			if (tokens.tokens[index].type == Token.Type.Keyword && tokens.tokens[index].token == "if" &&
677 				tokens.tokens[index+1].type == Token.Type.ParanthesesOpen){
678 				// now do the real work
679 				uinteger brackEnd = tokens.tokens.bracketPos(index+1);
680 				index += 2;
681 				ifNode.condition = generateCodeAST();
682 				// make sure index & brackEnd are now same
683 				if (index == brackEnd){
684 					index = brackEnd+1;
685 					ifNode.statement = generateStatementAST();
686 					ifNode.hasElse = false;
687 					// check if there's any else statement
688 					if (tokens.tokens[index].type == Token.Type.Keyword && tokens.tokens[index].token == "else"){
689 						// add that as well
690 						index ++;
691 						ifNode.elseStatement = generateStatementAST();
692 						ifNode.hasElse = true;
693 					}
694 				}else{
695 					compileErrors.append(CompileError(tokens.getTokenLine(index), "syntax error in condition"));
696 				}
697 			}else{
698 				compileErrors.append(CompileError(tokens.getTokenLine(index), "not a valid if/while statement"));
699 				index ++;

728 		/// returns a node representing either of the following:
729 		/// 
730 		/// 1. String literal
731 		/// 2. Number literal
732 		/// 3. Function Call (uses `generateFunctionCallAST`)
733 		/// 4. Variable (uses `generateVariableAST`)
734 		/// 5. Some code inside parantheses (uses `generateCodeAST`)
735 		/// 6. A literal array (`[x, y, z]`)
736 		/// 
737 		/// This function is used by `generateCodeAST` to separate nodes, and by `generateOperatorAST` to read operands
738 		CodeNode generateNodeAST(){
739 			Token token = tokens.tokens[index];
740 			// an identifier, or literal (i.e some data) was expected
741 			if (token.type == Token.Type.Identifier){
742 				if (index+1 < tokens.tokens.length && tokens.tokens[index+1].type == Token.Type.ParanthesesOpen){
743 					// is a function call
744 					CodeNode r = CodeNode(generateFunctionCallAST());
745 					// check if it skipped the semicolon, because if it did, it shouldnt have, so undo it
746 					if (tokens.tokens[index-1].type == Token.Type.StatementEnd){
747 						index --;
748 					}
749 					return r;
750 				}else{
751 					// just a var
752 					return CodeNode(generateVariableAST());
753 				}
754 			}else if (token.type == Token.Type.ParanthesesOpen){
755 				// some code
756 				return generateCodeAST();
757 			}else if (token.type == Token.Type.IndexBracketOpen){
758 				// literal array
759 				uinteger brackEnd = tokens.tokens.bracketPos(index);
760 				Token[] data = tokens.tokens[index .. brackEnd+1].dup;
761 				LiteralNode r;
762 				try{
763 					r.fromTokens(data);
764 				}catch (Exception e){
765 					compileErrors.append(CompileError(tokens.getTokenLine(index), e.msg));
766 					.destroy (e);
767 				}
768 				index = brackEnd+1;
769 				return CodeNode(r);
770 			}else if (token.type == Token.Type.Double || token.type == Token.Type.Integer || token.type == Token.Type.String){
771 				// literal
772 				index ++;
773 				try{
774 					return CodeNode(LiteralNode([token]));
775 				}catch (Exception e){
776 					compileErrors.append(CompileError(tokens.getTokenLine(index), e.msg));
777 					.destroy (e);
778 				}
779 			}else{
780 				index ++;
781 				compileErrors.append(CompileError(tokens.getTokenLine(index), "unexpected token"));
782 			}
783 			return CodeNode();
784 		}
