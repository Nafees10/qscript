25 /// generates html representation for ScriptNode
26 string[] toHtml(ScriptNode node){
27 	LinkedList!string html = new LinkedList!string;
28 	html.append(HTML_STYLE);
29 	html.append("<div>Script");
30 	foreach (functionDef; node.functions){
31 		html.append(functionDef.toHtml);
32 	}
33 	html.append("</div>");
34 	string[] r = html.toArray;
35 	.destroy(html);
36 	return r;
37 }
38 /// generates html representation for FunctionNode
39 string[] toHtml(FunctionNode node){
40 	LinkedList!string html = new LinkedList!string;
41 	html.append("<div>Function Definition<br>Name: "~node.name~"<br>Return Type: "~node.returnType.toString);
42 	// add the arguments
43 	if (node.arguments.length > 0){
44 		html.append("<div>Function Arguments");
45 		foreach (arg; node.arguments){

48 		}

76 /// generates html representation for StatementNode
77 string[] toHtml(StatementNode node){
78 	if (node.type == StatementNode.Type.Assignment){
79 		return node.node!(StatementNode.Type.Assignment).toHtml();
80 	}else if (node.type == StatementNode.Type.Block){
81 		return node.node!(StatementNode.Type.Block).toHtml();
82 	}else if (node.type == StatementNode.Type.FunctionCall){
83 		return node.node!(StatementNode.Type.FunctionCall).toHtml();
84 	}else if (node.type == StatementNode.Type.If){
85 		return node.node!(StatementNode.Type.If).toHtml();
86 	}else if (node.type == StatementNode.Type.While){
87 		return node.node!(StatementNode.Type.While).toHtml();
88 	}else if (node.type == StatementNode.Type.VarDeclare){
89 		return node.node!(StatementNode.Type.VarDeclare).toHtml();
90 	}else{
91 		throw new Exception("invalid stored type");
92 	}
93 }

172 /// generates html representation for CodeNode
173 string[] toHtml(CodeNode node, string caption = null){
174 	string[] r;
175 	if (node.type == CodeNode.Type.FunctionCall){
176 		r = node.node!(CodeNode.Type.FunctionCall).toHtml();
177 	}else if (node.type == CodeNode.Type.Literal){
178 		r = node.node!(CodeNode.Type.Literal).toHtml();
179 	}else if (node.type == CodeNode.Type.Operator){
180 		r = node.node!(CodeNode.Type.Operator).toHtml();
181 	}else if (node.type == CodeNode.Type.Variable){
182 		r = node.node!(CodeNode.Type.Variable).toHtml();
183 	}else{
184 		throw new Exception("invalid stored type");
185 	}
186 	if (caption != null){
187 		r = ["<div>"~caption]~r~"</div>";
188 	}
189 	return r;
190 }
