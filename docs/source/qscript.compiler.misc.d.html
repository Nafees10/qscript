35 /// used to store data types for data at compile time
36 package struct DataType{
37 	/// enum defining all data types
38 	public enum Type{
39 		Void,
40 		String,
41 		Integer,
42 		Double
43 	}
44 	/// the actual data type
45 	DataType.Type type;
46 	/// stores if it's an array. If type is `int`, it will be 0, if `int[]` it will be 1, if `int[][]`, then 2 ...
47 	uinteger arrayNestCount;
48 	/// returns true if it's an array
49 	@property bool isArray(){
50 		if (arrayNestCount > 0){
51 			return true;
52 		}
53 		return false;
54 	}
55 	/// constructor
56 	/// 
57 	/// dataType is the type to store
58 	/// arrayNest is the number of nested arrays
59 	this (DataType.Type dataType, uinteger arrayNest = 0){
60 		type = dataType;
61 		arrayNestCount = arrayNest;
62 	}
63 	/// constructor
64 	/// 
65 	/// `sType` is the type in string form
66 	this (string sType){
67 		fromString(sType);
68 	}
69 	/// constructor
70 	/// 
71 	/// `data` is the data to infer type from
72 	this (Token[] data){
73 		fromData(data);
74 	}
75 	/// reads DataType from a string, in case of failure or bad format in string, throws Exception
76 	void fromString(string s){
77 		string sType = null;
78 		uinteger indexCount = 0;
79 		// read the type
80 		for (uinteger i = 0, lastInd = s.length-1; i < s.length; i ++){
81 			if (s[i] == '['){
82 				sType = s[0 .. i];
83 				break;
84 			}else if (i == lastInd){
85 				sType = s;
86 				break;
87 			}
88 		}
89 		// now read the index
90 		for (uinteger i = sType.length; i < s.length; i ++){
91 			if (s[i] == '['){
92 				// make sure next char is a ']'
93 				i ++;
94 				if (s[i] != ']'){
95 					throw new Exception("invalid data type format");
96 				}
97 				indexCount ++;
98 			}else{
99 				throw new Exception("invalid data type");
100 			}
101 		}
102 		// now check if the type was ok or not
103 		if (sType == "void"){
104 			type = DataType.Type.Void;
105 		}else if (sType == "string"){
106 			type = DataType.Type.String;
107 		}else if (sType == "int"){
108 			type = DataType.Type.Integer;
109 		}else if (sType == "double"){
110 			type = DataType.Type.Double;
111 		}else{
112 			throw new Exception("invalid data type");
113 		}
114 		arrayNestCount = indexCount;
115 	}
116 
117 	/// identifies the data type from the actual data
118 	/// 
119 	/// throws Exception on failure
120 	void fromData(Token[] data){
121 		/// identifies type from data
122 		static DataType.Type identifyType(Token data){
123 			if (data.type == Token.Type.String){
124 				return DataType.Type.String;
125 			}else if (data.type == Token.Type.Integer){
126 				return DataType.Type.Integer;
127 			}else if (data.type == Token.Type.Double){
128 				return DataType.Type.Double;
129 			}else{
130 				throw new Exception("failed to read data type");
131 			}
132 		}
133 		// keeps count of number of "instances" of this function currently being executed
134 		static uinteger callCount = 0;
135 		callCount ++;
136 
137 		if (callCount == 1){
138 			this.arrayNestCount = 0;
139 			this.type = DataType.Type.Void;
140 		}
141 		// check if is an array
142 		if (data.length > 1 &&
143 			data[0].type == Token.Type.IndexBracketOpen && data[data.length-1].type == Token.Type.IndexBracketClose){
144 			// is an array
145 			this.arrayNestCount ++;
146 			// if elements are arrays, do recursion, else, just identify types
147 			Token[][] elements = splitArray(data);
148 			if (elements.length == 0){
149 				this.type = DataType.Type.Void;
150 			}else{
151 				// determine the type using recursion
152 				// stores the arrayNestCount till here
153 				uinteger thisNestCount = this.arrayNestCount;
154 				// stores the nestCount for the preious element, -1 if no previous element
155 				integer prevNestCount = -1;
156 				// stores the data type of the last element, void if no last element
157 				DataType.Type prevType = DataType.Type.Void;
158 				// now do recursion, and make sure all types come out same
159 				foreach (element; elements){
160 					fromData(element);
161 					// now make sure the nestCount came out same
162 					if (prevNestCount != -1){
163 						if (prevNestCount != this.arrayNestCount){
164 							throw new Exception("inconsistent data types in array elements");
165 						}
166 					}else{
167 						// set new nestCount
168 						prevNestCount = this.arrayNestCount;
169 					}
170 					// now to make sure type came out same
171 					if (prevType != DataType.Type.Void){
172 						if (this.type != prevType){
173 							throw new Exception("inconsistent data types in array elements");
174 						}
175 					}else{
176 						prevType = this.type;
177 					}
178 					// re-set the nestCount for the next element
179 					this.arrayNestCount = thisNestCount;
180 				}
181 				// now set the nestCount
182 				this.arrayNestCount = prevNestCount;
183 			}
184 		}else if (data.length == 0){
185 			this.type = DataType.Type.Void;
186 		}else{
187 			// then it must be only one token, if is zero, then it's void
188 			assert(data.length == 1, "non-array data must be only one token in length");
189 			// now check the type, and set it
190 			this.type = identifyType(data[0]);
191 		}
192 		callCount --;
193 	}
194 
195 	/// converts this DataType to string
196 	string toString(){
197 		char[] r;
198 		if (type == DataType.Type.Void){
199 			r = cast(char[]) "void";
200 		}else if (type == DataType.Type.Double){
201 			r = cast(char[]) "double";
202 		}else if (type == DataType.Type.Integer){
203 			r = cast(char[]) "int";
204 		}else if (type == DataType.Type.String){
205 			r = cast(char[]) "string";
206 		}else{
207 			throw new Exception("invalid type stored");
208 		}
209 		uinteger i = r.length;
210 		r.length += arrayNestCount * 2;
211 		for (; i < r.length; i += 2){
212 			r[i .. i+2] = "[]";
213 		}
214 		return cast(string) r;
215 	}
216 }

238 /// splits an array in tokens format to it's elements
239 /// 
240 /// For example, splitArray("[a, b, c]") will return ["a", "b", "c"]
241 package Token[][] splitArray(Token[] array){
242 	assert(array[0].type == Token.Type.IndexBracketOpen &&
243 		array[array.length - 1].type == Token.Type.IndexBracketClose, "not a valid array");
244 	LinkedList!(Token[]) elements = new LinkedList!(Token[]);
245 	for (uinteger i = 1, readFrom = i; i < array.length; i ++){
246 		// skip any other brackets
247 		if (array[i].type == Token.Type.BlockStart || array[i].type == Token.Type.IndexBracketOpen ||
248 			array[i].type == Token.Type.ParanthesesOpen){
249 			i = bracketPos!(true)(array, i);
250 			continue;
251 		}
252 		// check if comma is here
253 		if (array[i].type == Token.Type.Comma || array[i].type == Token.Type.IndexBracketClose){
254 			if ((readFrom > i || readFrom == i) && array[i].type == Token.Type.Comma){
255 				throw new Exception("syntax error");
256 			}
257 			elements.append(array[readFrom .. i]);
258 			readFrom = i + 1;
259 		}
260 	}
261 	Token[][] r = elements.toArray;
262 	.destroy(elements);
263 	return r;
264 }

267 ///

269 package integer strEnd(string s, uinteger i){
270 	for (i++;i<s.length;i++){
271 		if (s[i]=='\\'){
272 			i++;
273 			continue;
274 		}else if (s[i]=='"'){
275 			break;
276 		}
277 	}
278 	if (i==s.length){i=-1;}
279 	return i;

282 /// decodes a string. i.e, converts \t to tab, \" to ", etc
283 /// The string must not be surrounded by quoation marks
284 /// 
285 /// throws Exception on error
286 string decodeString(string s){
287 	string r;
288 	for (uinteger i = 0; i < s.length; i ++){
289 		if (s[i] == '\\'){
290 			// read the next char
291 			if (i == s.length-1){
292 				throw new Exception("unexpected end of string");
293 			}
294 			char nextChar = s[i+1];
295 			if (nextChar == '"'){
296 				r ~= '"';
297 			}else if (nextChar == 'n'){
298 				r ~= '\n';
299 			}else if (nextChar == 't'){
300 				r ~= '\t';
301 			}else if (nextChar == '\\'){
302 				r ~= '\\';
303 			}else{
304 				throw new Exception("\\"~nextChar~" is not an available character");
305 			}
306 			i ++;
307 			continue;
308 		}
309 		r ~= s[i];
310 	}
311 	return r;
312 }

314 /// encodes a string. i.e converts characters like tab, newline, to \t, \n ...
315 /// The string must not be enclosed in quotation marks
316 /// 
317 /// throws Exception on error
318 string encodeString(string s){
319 	string r;
320 	foreach (c; s){
321 		if (c == '\\'){
322 			r ~= "\\\\";
323 		}else if (c == '"'){
324 			r ~= "\\\"";
325 		}else if (c == '\n'){
326 			r ~= "\\n";
327 		}else if (c == '\t'){
328 			r ~= "\\t";
329 		}else{
330 			r ~= c;
331 		}
332 	}
333 	return r;
334 }

337 /// matches argument types with defined argument types. Used by ASTGen and compiler.d.
338 /// returns true if match successful, else, false
339 bool matchArguments(DataType[] definedTypes, DataType[] argTypes){
340 	if (argTypes.length != definedTypes.length){
341 		return false;
342 	}else{
343 		for (uinteger i = 0; i < argTypes.length; i ++){
344 			if (definedTypes[i].type == DataType.Type.Void){
345 				// skip checks
346 				continue;
347 			}else{
348 				if (argTypes[i].type != definedTypes[i].type){
349 					// check if is receiving a void[], against a someType[][].., then, it's ok
350 					if (argTypes[i].type == DataType.Type.Void &&
351 						argTypes[i].arrayNestCount > 0 && definedTypes[i].arrayNestCount > 0){
352 						return true;
353 					}
354 					return false;
355 				}
356 			}
357 			// check the array dimension
358 			if (definedTypes[i].arrayNestCount != argTypes[i].arrayNestCount){
359 				return false;
360 			}
361 		}
362 		return true;
363 	}
364 }

368 /// converts from Token[] to QData
369 /// 
370 /// `tokens` is the Token[] containing the data
371 /// `type` is the var in which the data type of the data will be put in
372 /// 
373 /// throws Exception on failure
374 package QData tokensToQData(Token[] tokens, ref DataType type){
375 	QData result;
376 	static uinteger callCount = 0;
377 	callCount ++;
378 	// check if is an array
379 	if (tokens.length > 1 &&
380 		tokens[0].type == Token.Type.IndexBracketOpen && tokens[tokens.length-1].type == Token.Type.IndexBracketClose){
381 
382 		// recursion...
383 		// or if is empty
384 		if (tokens.length == 2){
385 			// empty array
386 			type = DataType(DataType.Type.Void, 1);
387 			callCount --;
388 			return QData([]);
389 		}
390 		Token[][] elements = splitArray(tokens);
391 		result.arrayVal.length = elements.length;
392 		foreach (i, element; elements){
393 			result.arrayVal[i] = tokensToQData(element, type);
394 		}
395 		if (callCount == 1){
396 			type = DataType(tokens);
397 		}
398 	}else{
399 		type.fromData(tokens);
400 		assert(tokens.length == 1, "non-array data must be only one token in length");
401 		if (type.type == DataType.Type.Double){
402 			result.doubleVal = to!double(tokens[0].token);
403 		}else if (type.type == DataType.Type.Integer){
404 			result.intVal = to!integer(tokens[0].token);
405 		}else if (type.type == DataType.Type.String){
406 			assert(tokens[0].token.length > 1, "invalid string");
407 			result.strVal = decodeString(tokens[0].token[1 .. tokens[0].token.length - 1]);
408 		}
409 	}
410 	callCount --;
411 	return result;
412 }

431 /// converts a literal data in bytecode format into QData
432 /// 
433 /// throws Exception on error
434 package QData stringToQData(string literal){
435 	// make sure it's at least 2 in length
436 	if (literal.length < 2){
437 		throw new Exception("invalid byte code literal");
438 	}
439 	// check if it's an array
440 	if (literal[0] == '['){
441 		// get bracket end pos
442 		uinteger brackEnd = literal.bracketPos(0);
443 		// make sure endIndex is lastIndex
444 		if (brackEnd != literal.length-1){
445 			throw new Exception("invalid bracket-closing position in array");
446 		}
447 		LinkedList!QData elements = new LinkedList!QData;
448 		// use recursion to make that array
449 		for (uinteger i = 1, readFrom = 1; i <= brackEnd; i ++){
450 			if (literal[i] == '['){
451 				// skip to end
452 				i = literal.bracketPos(i);
453 			}else if (literal[i] == '"'){
454 				// skip the string
455 				i = literal.strEnd(i);
456 			}else if (literal[i] == ',' || i == brackEnd){
457 				// read the element
458 				if (readFrom < i){
459 					elements.append (stringToQData(literal[readFrom .. i]));
460 					readFrom = i+1;
461 				}
462 			}else if (literal[i] == ' ' || literal[i] == '\t'){
463 				// skip this char, if at the start/end of an element, otherwise, this aint allowed
464 				if (readFrom == i){
465 					readFrom ++;
466 				}else{
467 					throw new Exception("found whitespace at unexpected position");
468 				}
469 			}
470 		}
471 		// put em all in one QData
472 
473 		QData r = QData(elements.toArray);
474 		.destroy (elements);
475 		return r;
476 	}else{
477 		// must be a string, double, or int
478 		if (literal[0] == 's'){
479 			// string
480 			if (literal.length < 3){
481 				throw new Exception("invalid string in byte code literal");
482 			}
483 			if (literal[1] != '"'){
484 				throw new Exception("not a valid string");
485 			}
486 			// make sure strEnd == lastInd
487 			if (literal.strEnd(1) != literal.length -1){
488 				throw new Exception("quotation mark ended at wrong position in string");
489 			}
490 			// now just convert it
491 			return QData(decodeString(literal[2 .. literal.length - 1]));
492 		}else if (literal[0] == 'd'){
493 			// double
494 			return QData(to!double(literal[1 .. literal.length]));
495 		}else if (literal[0] == 'i'){
496 			return QData(to!integer(literal[1 .. literal.length]));
497 		}else{
498 			throw new Exception("invalid data type in byte code literal");
499 		}
500 	}
501 }

504 	assert ("i1".stringToQData == QData(integer(1)));

518 /// Each token is stored as a `Token` with the type and the actual token
519 package struct Token{
520 	/// Specifies type of token
521 	/// 
522 	/// used only in `compiler.tokengen`
523 	enum Type{
524 		String,/// That the token is: `"SOME STRING"`
525 		Integer,/// That the token an int
526 		Double, /// That the token is a double (floating point) value
527 		Identifier,/// That the token is an identifier. i.e token is a variable name or a function name.  For a token to be marked as Identifier, it doesn't need to be defined in `new()`
528 		DataType, /// the  token is a data type
529 		AssignmentOperator, /// and assignmentOperator
530 		Operator,/// That the token is an operator, like `+`, `==` etc
531 		Keyword,/// A `function` or `var` ...
532 		Comma,/// That its a comma: `,`
533 		StatementEnd,/// A semicolon
534 		ParanthesesOpen,/// `(`
535 		ParanthesesClose,/// `)`
536 		IndexBracketOpen,/// `[`
537 		IndexBracketClose,///`]`
538 		BlockStart,///`{`
539 		BlockEnd,///`}`
540 	}
541 	Type type;/// type of token
542 	string token;/// token
543 	this(Type tType, string tToken){
544 		type = tType;
545 		token = tToken;
546 	}
547 }
548 /// To store Tokens with Types where the line number of each token is required
549 package struct TokenList{
550 	Token[] tokens; /// Stores the tokens
551 	uinteger[] tokenPerLine; /// Stores the number of tokens in each line
552 	/// Returns the line number a token is in by usin the index of the token in `tokens`
553 	/// 
554 	/// The returned value is the line-number, not index, it starts from 1, not zero
555 	uinteger getTokenLine(uinteger tokenIndex){
556 		uinteger i = 0, chars = 0;
557 		tokenIndex ++;
558 		for (; chars < tokenIndex && i < tokenPerLine.length; i ++){
559 			chars += tokenPerLine[i];
560 		}
561 		return i;
562 	}
563 	/// reads tokens into a string
564 	string toString(Token[] t){
565 		char[] r;
566 		// set length
567 		uinteger length = 0;
568 		for (uinteger i = 0; i < t.length; i ++){
569 			length += t[i].token.length;
570 		}
571 		r.length = length;
572 		// read em all into r;
573 		for (uinteger i = 0, writeTo = 0; i < t.length; i ++){
574 			r[writeTo .. writeTo + t[i].token.length] = t[i].token;
575 			writeTo += t[i].token.length;
576 		}
577 		return cast(string)r;
578 	}
579 }

581 /// Checks if the brackets in a tokenlist are in correct order, and are closed
582 /// 
583 /// In case not, returns false, and appends error to `errorLog`
584 package bool checkBrackets(TokenList tokens, LinkedList!CompileError errors){
585 	enum BracketType{
586 		Round,
587 		Square,
588 		Block
589 	}
590 	BracketType[Token.Type] brackOpenIdent = [
591 		Token.Type.ParanthesesOpen: BracketType.Round,
592 		Token.Type.IndexBracketOpen: BracketType.Square,
593 		Token.Type.BlockStart: BracketType.Block
594 	];
595 	BracketType[Token.Type] brackCloseIdent = [
596 		Token.Type.ParanthesesClose: BracketType.Round,
597 		Token.Type.IndexBracketClose: BracketType.Square,
598 		Token.Type.BlockEnd:BracketType.Block
599 	];
600 	Stack!BracketType bracks = new Stack!BracketType;
601 	Stack!uinteger bracksStartIndex = new Stack!uinteger;
602 	BracketType curType;
603 	bool r = true;
604 	for (uinteger lastInd = tokens.tokens.length-1, i = 0; i<=lastInd; i++){
605 		if (tokens.tokens[i].type in brackOpenIdent){
606 			bracks.push(brackOpenIdent[tokens.tokens[i].type]);
607 			bracksStartIndex.push(i);
608 		}else if (tokens.tokens[i].type in brackCloseIdent){
609 			bracksStartIndex.pop();
610 			if (bracks.pop != brackCloseIdent[tokens.tokens[i].type]){
611 				errors.append(CompileError(tokens.getTokenLine(i),
612 						"brackets order is wrong; first opened must be last closed"));
613 				r = false;
614 				break;
615 			}
616 		}else if (i == lastInd && bracks.count > 0){
617 			// no bracket ending
618 			i = -2;
619 			errors.append(CompileError(tokens.getTokenLine(bracksStartIndex.pop), "bracket not closed"));
620 			r = false;
621 			break;
622 		}
623 	}
624 
625 	.destroy(bracks);
626 	.destroy(bracksStartIndex);
627 	return r;
628 }

630 /// Returns index of closing/openinig bracket of the provided bracket  
631 /// 
632 /// `forward` if true, then the search is in forward direction, i.e, the closing bracket is searched for
633 /// `tokens` is the array of tokens to search in
634 /// `index` is the index of the opposite bracket
635 /// 
636 /// It only works correctly if the brackets are in correct order, and the closing bracket is present  
637 /// so, before calling this, `compiler.misc.checkBrackets` should be called
638 package uinteger bracketPos(bool forward=true)(Token[] tokens, uinteger index){
639 	Token.Type[] closingBrackets = [
640 		Token.Type.BlockEnd,
641 		Token.Type.IndexBracketClose,
642 		Token.Type.ParanthesesClose
643 	];
644 	Token.Type[] openingBrackets = [
645 		Token.Type.BlockStart,
646 		Token.Type.IndexBracketOpen,
647 		Token.Type.ParanthesesOpen
648 	];
649 	uinteger count; // stores how many closing/opening brackets before we reach the desired one
650 	uinteger i = index;
651 	for (uinteger lastInd = (forward ? tokens.length : 0); i != lastInd; (forward ? i ++: i --)){
652 		if ((forward ? openingBrackets : closingBrackets).hasElement(tokens[i].type)){
653 			count ++;
654 		}else if ((forward ? closingBrackets : openingBrackets).hasElement(tokens[i].type)){
655 			count --;
656 		}
657 		if (count == 0){
658 			break;
659 		}
660 	}
661 	return i;
662 }
663 ///
664 unittest{
665 	Token[] tokens;
666 	tokens = [
667 		Token(Token.Type.Comma, ","), Token(Token.Type.BlockStart,"{"), Token(Token.Type.Comma,","),
668 		Token(Token.Type.IndexBracketOpen,"["),Token(Token.Type.IndexBracketClose,"]"),Token(Token.Type.BlockEnd,"}")
669 	];
670 	assert(tokens.bracketPos!true(1) == 5);
671 	assert(tokens.bracketPos!false(5) == 1);
672 	assert(tokens.bracketPos!true(3) == 4);
673 	assert(tokens.bracketPos!false(4) == 3);
674 }

676 /// Returns index of closing/openinig bracket of the provided bracket  
677 /// 
678 /// `forward` if true, then the search is in forward direction, i.e, the closing bracket is searched for
679 /// `s` is the string to search in
680 /// `index` is the index of the opposite bracket
681 /// 
682 /// throws Exception if the bracket is not found
683 package uinteger bracketPos(bool forward=true)(string s, uinteger index){
684 	char[] closingBrackets = [']','}',')'];
685 	char[] openingBrackets = ['[','{','('];
686 	Stack!char brackets = new Stack!char;
687 	uinteger i = index;
688 	for (uinteger lastInd = (forward ? s.length : 0); i != lastInd; (forward ? i ++: i --)){
689 		if ((forward ? openingBrackets : closingBrackets).hasElement(s[i])){
690 			// push it to brackets
691 			brackets.push(s[i]);
692 		}else if ((forward ? closingBrackets : openingBrackets).hasElement(s[i])){
693 			// make sure the correct bracket was closed
694 			char opposite = brackets.pop;
695 			if ((forward ? openingBrackets : closingBrackets).indexOf(s[i]) !=
696 				(forward ? closingBrackets : openingBrackets).indexOf(opposite)){
697 				throw new Exception("incorect brackets order - first opened must be last closed");
698 			}
699 		}
700 		if (brackets.count == 0){
701 			break;
702 		}
703 	}
704 	.destroy (brackets);
705 	return i;
706 }
707 ///
708 unittest{
709 	assert ("hello(asdf[asdf])".bracketPos(5) == 16);
710 	assert ("hello(asdf[asdf])".bracketPos(10) == 15);
711 }

713 /// removes "extra" whitespace from a string. i.e, if there are more than 1 consecutive spaces/tabs, one is removed
714 /// 
715 /// `line` is the string to remove whitespace from
716 /// `commentStart` is the character that marks the start of a comment, if ==0, then comments are not not considered
717 package string removeWhitespace(string line, char commentStart=0){
718 	bool prevWasWhitespace = true;
719 	string r;
720 	foreach (c; line){
721 		if (prevWasWhitespace && (c == ' ' || c == '\t')){
722 			// do not add this one then
723 			continue;
724 		}else if (c != 0 && c == commentStart){
725 			break;
726 		}else{
727 			prevWasWhitespace = false;
728 			r ~= c;
729 		}
730 	}
731 	return r;
732 }
