85 /// a node to represent code that evaluates to some data.
86 /// 
87 /// This node can contain:  
88 /// 1. Function call - to only those functions that return some data
89 /// 2. Literals
90 /// 3. Operators
91 /// 4. Variables
92 package struct CodeNode{
93 	/// enum defining the possible types this node can store
94 	public enum Type{
95 		FunctionCall,
96 		Literal,
97 		Operator,
98 		Variable,
99 		ReadElement
100 	}
101 	/// the stored type
102 	private Type storedType;
103 	/// union storing all possible nodes
104 	private union{
105 		FunctionCallNode fCall;
106 		LiteralNode literal;
107 		OperatorNode operator;
108 		VariableNode var;
109 		ReadElement arrayRead;
110 	}
111 	/// returns the type of the stored type
112 	@property CodeNode.Type type(){
113 		return storedType;
114 	}
115 	/// returns the data type that the node will return
116 	@property DataType returnType(){
117 		if (storedType == CodeNode.Type.FunctionCall){
118 			return fCall.returnType;
119 		}else if (storedType == CodeNode.Type.Literal){
120 			return literal.type;
121 		}else if (storedType == CodeNode.Type.Operator){
122 			return operator.returnType;
123 		}else if (storedType == CodeNode.Type.Variable){
124 			return var.varType;
125 		}else if (storedType == CodeNode.Type.ReadElement){
126 			return arrayRead.returnType;
127 		}else{
128 			throw new Exception("invalid type stored");
129 		}
130 	}
131 	/// sets the stored node
132 	@property auto node (T)(T newNode){
133 		static if (is (T == FunctionCallNode)){
134 			storedType = CodeNode.Type.FunctionCall;
135 			return fCall = newNode;
136 		}else static if (is (T == LiteralNode)){
137 			storedType = CodeNode.Type.Literal;
138 			return literal = newNode;
139 		}else static if (is (T == OperatorNode)){
140 			storedType = CodeNode.Type.Operator;
141 			return operator = newNode;
142 		}else static if (is (T == VariableNode)){
143 			storedType = CodeNode.Type.Variable;
144 			return var = newNode;
145 		}else static if (is (T == ReadElement)){
146 			storedType = CodeNode.Type.ReadElement;
147 			return arrayRead = newNode;
148 		}
149 	}
150 	/// returns the stored type
151 	@property auto node(CodeNode.Type T)(){
152 		// make sure it's the correct type
153 		if (T != storedType){
154 			throw new Exception("attempting to retrieve invalid type from CodeNode.node");
155 		}
156 		static if (T == CodeNode.Type.FunctionCall){
157 			return fCall;
158 		}else static if (T == CodeNode.Type.Literal){
159 			return literal;
160 		}else static if (T == CodeNode.Type.Operator){
161 			return operator;
162 		}else static if (T == CodeNode.Type.Variable){
163 			return var;
164 		}else static if (T == CodeNode.Type.ReadElement){
165 			return arrayRead;
166 		}else{
167 			throw new Exception("attempting to retrieve invalid type from CodeNode.node");
168 		}
169 	}
170 	/// constructor
171 	this (T)(T newNode){
172 		node = newNode;
173 	}
174 }

194 /// stores literal data, i.e data that was availabe at runtime. Can store strings, double, integer, array
195 package struct LiteralNode{
196 	private import qscript.qscript : QData;
197 	/// stores the data type for the literal
198 	public DataType type;
199 	/// stores the literal in a QData
200 	public QData literal;
201 	/// constructor
202 	this (QData data, DataType dataType){
203 		literal = data;
204 	}
205 	/// constructor using `fromTokens`
206 	this (Token[] tokensLiteral){
207 		fromTokens(tokensLiteral);
208 	}
209 	/// reads the literal from a string
210 	/// 
211 	/// throws Exception on error
212 	void fromTokens(Token[] tokensLiteral){
213 		// check if is empty array
214 		literal = tokensToQData(tokensLiteral, type);
215 	}
216 	/// returns this literal as in a string-representation, for the bytecode
217 	string toByteCode(){
218 		/// returns array in byte code representation
219 		static string fromDataOrArray(QData data, DataType type){
220 			if (type.arrayNestCount > 0){
221 				char[] array = ['['];
222 				// the type of the elements
223 				DataType subType = type;
224 				subType.arrayNestCount --;
225 				// use recursion
226 				foreach (element; data.arrayVal){
227 					array ~= cast(char[])fromDataOrArray(element, subType) ~ ',';
228 				}
229 				if (array.length == 1){
230 					array ~= ']';
231 				}else{
232 					array[array.length - 1] = ']';
233 				}
234 				return cast(string)array;
235 			}else{
236 				if (type.type == DataType.Type.Double){
237 					return "d"~to!string(data.doubleVal);
238 				}else if (type.type == DataType.Type.Integer){
239 					return "i"~to!string(data.intVal);
240 				}else if (type.type == DataType.Type.String){
241 					return "s\""~encodeString(data.strVal)~'"';
242 				}else{
243 					throw new Exception("invalid type stored");
244 				}
245 			}
246 		}
247 		return fromDataOrArray(literal, type);
248 	}
249 }

280 /// stores an "array-read" (instruction `readElement` or for string, `readChar`)
281 package struct ReadElement{
282 	/// stores the nodes. [0] is the node to read from. [1] is the index
283 	private CodeNode[] nodes = [null, null];
284 	/// the node to read from
285 	public @property CodeNode readFromNode(){
286 		return nodes[0];
287 	}
288 	/// the node to read from
289 	public @property CodeNode readFromNode(CodeNode newNode){
290 		return nodes[0] = newNode;
291 	}
292 	/// the index to read at
293 	public @property CodeNode index(){
294 		return nodes[1];
295 	}
296 	/// the index to read at
297 	public @property CodeNode index(CodeNode newNode){
298 		return nodes[1] = newNode;
299 	}
300 	/// returns the data type this node will return
301 	public @property DataType returnType(){
302 		DataType r = nodes[0].returnType;
303 		if (r.arrayNestCount == 0 && r.type == DataType.Type.String){
304 			return DataType(DataType.Type.String);
305 		}
306 		r.arrayNestCount --;
307 		return r;
308 	}
309 	/// constructor
310 	this (CodeNode toReadNode, CodeNode index){
311 		nodes.length = 2;
312 		nodes[0] = toReadNode;
313 		nodes[1] = index;
314 	}

317 /// a node representing statements, including: if, while, function-call..
318 package struct StatementNode{
319 	/// types of a statement
320 	public enum Type{
321 		If,
322 		While,
323 		Block,
324 		Assignment,
325 		FunctionCall,
326 		VarDeclare
327 	}
328 	/// type of this statement
329 	private Type storedType;
330 	/// the stored node, is in this union
331 	private union{
332 		IfNode ifNode;
333 		WhileNode whileNode;
334 		BlockNode blockNode;
335 		FunctionCallNode functionCallNode;
336 		VarDeclareNode varDeclareNode;
337 		AssignmentNode assignNode;
338 	}
339 	/// modifies the stored node
340 	@property node(T)(T newNode){
341 		static if (is (T == IfNode)){
342 			storedType = StatementNode.Type.If;
343 			ifNode = newNode;
344 		}else static if (is (T == WhileNode)){
345 			storedType = StatementNode.Type.While;
346 			whileNode = newNode;
347 		}else static if (is (T == BlockNode)){
348 			storedType = StatementNode.Type.Block;
349 			blockNode = newNode;
350 		}else static if (is (T == FunctionCallNode)){
351 			storedType = StatementNode.Type.FunctionCall;
352 			functionCallNode = newNode;
353 		}else static if (is (T == VarDeclareNode)){
354 			storedType = StatementNode.Type.VarDeclare;
355 			varDeclareNode = newNode;
356 		}else if (is (T == AssignmentNode)){
357 			storedType = StatementNode.Type.Assignment;
358 			assignNode = newNode;
359 		}else{
360 			throw new Exception("attempting to assign invalid node type to StatementNode.node");
361 		}
362 	}
363 	/// returns the stored type
364 	@property node(StatementNode.Type T)(){
365 		// make sure type is correct
366 		if (T != storedType){
367 			throw new Exception("stored type does not match with type being retrieved");
368 		}
369 		static if (T == StatementNode.Type.If){
370 			return ifNode;
371 		}else static if (T == StatementNode.Type.While){
372 			return whileNode;
373 		}else static if (T == StatementNode.Type.Block){
374 			return blockNode;
375 		}else static  if (T == StatementNode.Type.FunctionCall){
376 			return functionCallNode;
377 		}else static if (T == StatementNode.Type.VarDeclare){
378 			return varDeclareNode;
379 		}else if (T == StatementNode.Type.Assignment){
380 			return assignNode;
381 		}else{
382 			throw new Exception("attempting to retrieve invalid type from StatementNode.node");
383 		}
384 	}
385 	/// returns the type of the stored node
386 	@property StatementNode.Type type(){
387 		return storedType;
388 	}
389 	/// constructor
390 	this (T)(T newNode){
391 		node = newNode;
392 	}
