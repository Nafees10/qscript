16 	/// generates byte code for a script
17 	public string[] generateByteCode(ScriptNode scriptNode, ref CompileError[] errors){
18 		compileErrors = new LinkedList!CompileError;
19 		LinkedList!string byteCode = new LinkedList!string;
20 		// remove all previous errors
21 		compileErrors.clear;
22 		// prepare a list of script-defind functions
23 		scriptDefinedFunctions.length = scriptNode.functions.length;
24 		foreach (i, functionNode; scriptNode.functions){
25 			scriptDefinedFunctions[i] = functionNode.name;
26 		}
27 		// go through all nodes/ functions, and generate byte-code for them
28 		foreach (functionNode; scriptNode.functions){
29 			byteCode.append(generateByteCode(functionNode));
30 		}
31 
32 		string[] r = byteCode.toArray;
33 		.destroy(byteCode);
34 		// put in errors
35 		errors = compileErrors.toArray;
36 		.destroy(compileErrors);
37 		return r;
38 	}

41 	private struct{
42 		/// stores IDs for vars
43 		private string[uinteger] varIDs;
44 		/// stores the scope "depth" for each varID
45 		private uinteger[uinteger] varScopeDepth;
46 		/// stores the current scope depth
47 		private uinteger scopeDepth;
48 		/// creates a new var, returns the ID, throws Exception if var already exists
49 		public uinteger getNewVarID(string varName){
50 			// check if already exists
51 			foreach (id, name; varIDs){
52 				if (varName == name){
53 					throw new Exception("variable '"~varName~"' declared twice");
54 				}
55 			}
56 			// find an empty ID
57 			uinteger newID = 0;
58 			while (newID in varIDs){
59 				newID ++;
60 			}
61 			varIDs[newID] = varName;
62 			// put it in varScopeDepth
63 			varScopeDepth[newID] = scopeDepth;
64 			return newID;
65 		}
66 		/// returns ID of a var. throws exception if var is not declared
67 		public uinteger getVarID(string varName){
68 			foreach (key; varIDs.keys){
69 				if (varIDs[key] == varName){
70 					return key;
71 				}
72 			}
73 			throw new Exception("variable '"~varName~"' not declared but used");
74 		}
75 		/// should/must be called when the next few calls of var-related functions will be from inside another block
76 		public void increaseScope(){
77 			scopeDepth ++;
78 		}
79 		/// should/must be called when all var related functions from inside a block are done
80 		public void decreaseScope(){
81 			// clear vars
82 			foreach (key; varIDs.keys){
83 				if (varScopeDepth[key] == scopeDepth){
84 					varScopeDepth.remove(key);
85 					varIDs.remove(key);
86 				}
87 			}
88 			if (scopeDepth > 0){
89 				scopeDepth --;
90 			}
91 		}
92 	}

107 	/// returns the number of max vars that exist at one time in a StatementNode[]
108 	private static uinteger getMaxVarCount(StatementNode[] statements){
109 		uinteger r = 0;
110 		uinteger maxSub = 0;
111 		foreach (statement; statements){
112 			if (statement.type == StatementNode.Type.VarDeclare){
113 				// get the number of vars
114 				r += statement.node!(StatementNode.Type.VarDeclare).vars.length;
115 			}else{
116 				uinteger newMaxSub = getVarCountStatement (statement);
117 				if (newMaxSub > maxSub){
118 					maxSub = newMaxSub;
119 				}
120 			}
121 		}
122 		return r + maxSub;
123 	}

125 	/// returns the number of vars defined in a statement
126 	private static uinteger getVarCountStatement (StatementNode statement){
127 		if (statement.type == StatementNode.Type.VarDeclare){
128 			return statement.node!(StatementNode.Type.VarDeclare).vars.length;
129 		}else if (statement.type == StatementNode.Type.Block){
130 			return getMaxVarCount (statement.node!(StatementNode.Type.Block).statements);
131 		}else if (statement.type == StatementNode.Type.If){
132 			uinteger onTrueCount = 0, onFalseCount = 0;
133 			IfNode ifStatement = statement.node!(StatementNode.Type.If);
134 			onTrueCount = getVarCountStatement (ifStatement.statement);
135 			if (ifStatement.hasElse){
136 				onFalseCount = getVarCountStatement (ifStatement.elseStatement);
137 			}
138 			return (onTrueCount > onFalseCount ? onTrueCount : onFalseCount);
139 		}else if (statement.type == StatementNode.Type.While){
140 			return getVarCountStatement (statement.node!(StatementNode.Type.While).statement);
141 		}
142 		return 0;
143 	}

148 	/// generates byte code for FunctionNode
149 	private string[] generateByteCode(FunctionNode node){
150 		LinkedList!string byteCode = new LinkedList!string;
151 		// push the function name
152 		byteCode.append (node.name);
153 		// then set the max var count
154 		uinteger maxVars = getMaxVarCount(node.bodyBlock.statements);
155 		// add the arguments count to this
156 		maxVars += node.arguments.length;
157 		byteCode.append("\tvarCount i"~to!string(maxVars));
158 		// then assign the arguments their var IDs
159 		increaseScope();
160 		foreach (arg; node.arguments){
161 			try{
162 				getNewVarID(arg.argName);
163 			}catch (Exception e){
164 				compileErrors.append(CompileError(0, "bug in compiler - argument name '"~arg.argName~"' not available"));
165 				.destroy (e);
166 			}
167 		}
168 		// then append the instructions
169 		byteCode.append (generateByteCode (node.bodyBlock));
170 		decreaseScope();
171 
172 		string[] r = byteCode.toArray;
173 		.destroy(byteCode);
174 		return r;
175 	}

190 	/// generates byte code for StatementNode
191 	private string[] generateByteCode(StatementNode node){
192 		if (node.type == StatementNode.Type.Assignment){
193 			return generateByteCode(node.node!(StatementNode.Type.Assignment));
194 		}else if (node.type == StatementNode.Type.Block){
195 			return generateByteCode(node.node!(StatementNode.Type.Block));
196 		}else if (node.type == StatementNode.Type.FunctionCall){
197 			return generateByteCode(node.node!(StatementNode.Type.FunctionCall));
198 		}else if (node.type == StatementNode.Type.If){
199 			return generateByteCode(node.node!(StatementNode.Type.If));
200 		}else if (node.type == StatementNode.Type.While){
201 			return generateByteCode(node.node!(StatementNode.Type.While));
202 		}else if (node.type == StatementNode.Type.VarDeclare){
203 			// VarDeclare wont be converted to anything, these only exist to tell FunctionNode the varCount
204 			return generateByteCode(node.node!(StatementNode.Type.VarDeclare));
205 		}else{
206 			compileErrors.append(CompileError(0, "invalid AST generated"));
207 			return [];
208 		}

211 	/// generates byte code for FunctionCallNode
212 	private string[] generateByteCode(FunctionCallNode node, bool pushReturn = false){
213 		auto byteCode = new LinkedList!string;
214 		// push the args
215 		uinteger argCount = 0;
216 		foreach (arg; node.arguments){
217 			byteCode.append(generateByteCode(arg));
218 			argCount ++;
219 		}
220 		// check if is a predefined-QScript-function
221 		const string[string] predefinedFunctions = [
222 			"setLength"		: "setLen",
223 			"getLength"		: "getLen",
224 			"strLen"		: "strLen",
225 			"array"			: "makeArray",
226 			"return"		: "return",
227 			"strToInt"		: "strToInt",
228 			"strToDouble"	: "strToDouble",
229 			"intToStr"		: "intToStr",
230 			"intToDouble"	: "intToDouble",
231 			"doubleToStr"	: "doubleToStr",
232 			"doubleToInt"	: "doubleToInt",
233 			"not"			: "not"
234 		];
235 		if (node.fName in predefinedFunctions){
236 			// then use the predefined function's instruction
237 			if (node.fName == "array"){
238 				byteCode.append ('\t'~predefinedFunctions[node.fName]~" i"~to!string(node.arguments.length));
239 			}else{
240 				byteCode.append('\t'~predefinedFunctions[node.fName]);
241 			}
242 		}else{
243 			// then append the instruction to execute this function
244 			if (isScriptDefined(node.fName)){
245 				byteCode.append("\texecFuncS s\""~node.fName~"\" i"~to!string(argCount));
246 			}else{
247 				byteCode.append("\texecFuncE s\""~node.fName~"\" i"~to!string(argCount));
248 			}
249 		}
250 		// pop the return, if result not required
251 		if (!pushReturn){
252 			byteCode.append("\tpop i1");
253 		}
254 		string[] r = byteCode.toArray;
255 		.destroy(byteCode);
256 		return r;
257 	}

277 	/// generates byte code for OperatorNode
278 	private string[] generateByteCode(OperatorNode node){
279 		const string[string] operatorInstructions = [
280 			"*" : "multiply",
281 			"/" : "divide",
282 			"+" : "add",
283 			"-" : "subtract",
284 			"%" : "mod",
285 			"~" : "concat",
286 			"<" : "isLesser",
287 			"==": "isSame",
288 			">" : "isGreater",
289 			"&&": "and",
290 			"||": "or"
291 		];
292 		// check if types are ok
293 		const static string[] ArtithmaticOperators = ["*", "/", "+", "-", "%", "<", ">"];
294 		const static string[] IntOperators = ["&&", "||"];
295 		const static string[] ArrayStringOperators = ["~"];
296 		const static string[] NonArrayOperators = ["==", "&&", "||"];
297 		bool errorFree = true;
298 		// first make sure both operands are of same type
299 		if (node.operands[0].returnType != node.operands[1].returnType){
300 			compileErrors.append(CompileError(0, "data type of both operands must be same"));
301 		}
302 		if (ArtithmaticOperators.hasElement(node.operator)){
303 			// must be either int, or double
304 			if (node.operands[0].returnType.isArray == true){
305 				compileErrors.append(CompileError(0, "arithmatic operators cannot be used on arrays"));
306 				errorFree = false;
307 			}
308 			if (node.operands[0].returnType.type != DataType.Type.Double &&
309 				node.operands[0].returnType.type != DataType.Type.Integer){
310 				compileErrors.append(CompileError(0, "arithmatic operators can only be used on integer or double"));
311 				errorFree = false;
312 			}
313 		}else if (IntOperators.hasElement(node.operator)){
314 			// make sure is int
315 			if (node.operands[0].returnType != DataType(DataType.Type.Integer)){
316 				compileErrors.append(CompileError(0, node.operator~" can only be used on integers"));
317 				errorFree = false;
318 			}
319 		}else if (ArrayStringOperators.hasElement(node.operator)){
320 			if (!node.operands[0].returnType.isArray && node.operands[0].returnType.type != DataType.Type.String){
321 				compileErrors.append(CompileError(0, "cannot use this operator on non-string and non-array data"));
322 				errorFree = false;
323 			}
324 		}else if (NonArrayOperators.hasElement(node.operator)){
325 			// make sure type's not array
326 			if (node.operands[0].returnType.isArray){
327 				compileErrors.append(CompileError(0, node.operator~" cannot be used with arrays"));
328 				errorFree = false;
329 			}
330 		}else{
331 			compileErrors.append(CompileError(0, "invalid operator"));
332 			errorFree = false;
333 		}
334 		if (errorFree){
335 			// now push the operands
336 			auto byteCode = new LinkedList!string;
337 			byteCode.append(generateByteCode(node.operands[0])~generateByteCode(node.operands[1]));
338 			// now get the operator instruction name
339 			string instruction = operatorInstructions[node.operator];
340 			// append the data type at the end to get the exact name, only if is not IntOperator
341 			if (!IntOperators.hasElement(node.operator)){
342 				if (node.operands[0].returnType.isArray){
343 					instruction ~= "Array";
344 				}else if (node.operands[0].returnType.type == DataType.Type.String){
345 					instruction ~= "String";
346 				}else if (node.operands[0].returnType.type == DataType.Type.Integer){
347 					instruction ~= "Int";
348 				}else if (node.operands[0].returnType.type == DataType.Type.Double){
349 					instruction ~= "Double";
350 				}else{
351 					compileErrors.append(CompileError(0, "unsupported data type"));
352 				}
353 			}
354 			// call the instruction
355 			byteCode.append("\t"~instruction);
356 			string[] r = byteCode.toArray;
357 			.destroy(byteCode);
358 			return r;
359 		}else{
360 			return [];
361 		}
362 	}

391 	/// generates byte code for AssignmentNode
392 	private string[] generateByteCode(AssignmentNode node){
393 		uinteger varID;
394 		try{
395 			varID = getVarID(node.var.varName);
396 		}catch (Exception e){
397 			compileErrors.append(CompileError(0, e.msg));
398 			.destroy(e);
399 		}
400 		// first check if the var is an array
401 		if (node.indexes.length > 0){
402 			// push the original value of var, then push all the indexes, then call modifyAray, and setVar
403 			auto byteCode = new LinkedList!string;
404 			byteCode.append("\tgetVar i"~to!string(varID));
405 			// now push the val
406 			byteCode.append(generateByteCode(node.val));
407 			// now the indexes
408 			foreach (index; node.indexes){
409 				byteCode.append(generateByteCode(index));
410 			}
411 			// then modifyArray
412 			byteCode.append("\tmodifyArray i"~to!string(node.indexes.length));
413 			// finally, set the new array back
414 			byteCode.append("\tsetVar i"~to!string(varID));
415 			string[] r = byteCode.toArray;
416 			.destroy(byteCode);
417 			return r;
418 		}else{
419 			// just push the val, and call setVar on the varID
420 			return generateByteCode(node.val)~
421 			["\tsetVar i"~to!string(varID)];
422 		}
423 	}

428 		foreach (var; node.vars){

431 			}catch (Exception e){
432 				compileErrors.append(CompileError(0, e.msg));
433 			}
434 		}
435 		// if is an array, then call emptyArray on all the vars
436 		if (node.type.isArray){
437 			const string makeInst = "\temptyArray";
438 			string[] r;
439 			r.length = 2*node.vars.length;
440 			foreach (i, var; node.vars){
441 				r[i*2] = makeInst;
442 				r[(i*2)+1] = "\tsetVar i"~to!string(getVarID(var));
443 			}
444 			return r;
445 		}
446 		return [];

449 	/// generates byte code for IfNode
450 	private string[] generateByteCode(IfNode node){
451 		// stores the `ID` of this if statement
452 		static uinteger ifCount = 0;
453 		uinteger currentCount = ifCount;
454 		ifCount ++;
455 		auto byteCode = new LinkedList!string;
456 		// first push the condition
457 		byteCode.append(generateByteCode(node.condition));
458 		// then the skipTrue, to skip the jump to else
459 		byteCode.append([
460 				"\tskipTrue i1",
461 				"\tjump ifEnd"~to!string(currentCount)
462 			]);
463 		// then comes the if-on-true body
464 		byteCode.append(generateByteCode(node.statement));
465 		// then then jump to elseEnd to skip the else statements
466 		if (node.hasElse){
467 			byteCode.append("\tjump elseEnd"~to!string(currentCount));
468 		}
469 		byteCode.append("\tifEnd"~to!string(currentCount)~":");
470 		// now the elseBody
471 		if (node.hasElse){
472 			byteCode.append(generateByteCode(node.elseStatement)~
473 					["\telseEnd"~to!string(currentCount)~":"]);
474 		}
475 		string[] r = byteCode.toArray;
476 		.destroy(byteCode);
477 		return r;
478 	}

481 	private string[] generateByteCode(WhileNode node){
482 		auto byteCode = new LinkedList!string;
483 		// stores the `ID` of the while statement
484 		static uinteger whileCount = 0;
485 		uinteger currentCount = whileCount;
486 		whileCount ++;
487 		// first comes the jump-back-here position
488 		byteCode.append("\twhileStart"~to!string(currentCount)~":");
489 		// then comes the condition
490 		byteCode.append(generateByteCode(node.condition));
491 		// then skip the jump-to-end if true
492 		byteCode.append([
493 				"\tskipTrue i1",
494 				"\tjump whileEnd"~to!string(currentCount)
495 			]);
496 		// then the loop body
497 		byteCode.append(generateByteCode(node.statement));
498 		// then jump back to condition, to see if it'll start again
499 		byteCode.append("\tjump whileStart"~to!string(currentCount));
500 		// then mark the loop end
501 		byteCode.append("\twhileEnd"~to!string(currentCount)~":");
502 		string[] r = byteCode.toArray;
503 		.destroy(byteCode);
504 		return r;
