32 	/// reads byte code from string[] into Function[] structs
33 	public static Function[] readByteCode (string[] byteCode){
34 		LinkedList!Instruction instructions = new LinkedList!Instruction;
35 		LinkedList!Function functions = new LinkedList!Function;
36 		string fName = null;
37 
38 		for (uinteger i = 0, readFrom = 0, lastind = byteCode.length-1; i < byteCode.length; i ++){
39 			string[] lineWords = byteCode[i].readWords;
40 			if (lineWords.length == 0){
41 				continue;
42 			}
43 			// if only one word && byteCode[i][0]==space/tab, then it's a functionDef start
44 			if (lineWords.length == 1 && byteCode[i][0] != ' ' && byteCode[i][0] != '\t'){
45 				// new function start
46 				if (fName != null){
47 					// add previous function
48 					functions.append (Function(fName, instructions.toArray));
49 				}
50 				instructions.clear;
51 				fName = lineWords[0];
52 			}else{
53 				// probably a statement
54 				Instruction inst;
55 				if (lineWords.length > 1){
56 					inst = Instruction(lineWords[0], lineWords[1 .. lineWords.length].dup);
57 				}else{
58 					inst = Instruction(lineWords[0], []);
59 				}
60 				instructions.append(inst);
61 			}
62 			// if lastInd, append the last function
63 			if (i == lastind){
64 				functions.append (Function(fName, instructions.toArray));
65 				instructions.clear;
66 			}
67 		}
68 		.destroy(instructions);
69 		Function[] r = functions.toArray;
70 		.destroy(functions);
71 		return r;
72 	}
73 	/// converts byte code into array of ByteCodeFunction
74 	public static ByteCodeFunction!(T)[] toFunctionPtr(T)(string[] byteCode, T[string] instructionsPtr){
75 		/// searches for jump positions, returns assoc_array containing their indexes
76 		static uinteger[string] searchJumpPos(Instruction[] instructions){
77 			uinteger[string] r;
78 			uinteger instructionCount = 0;
79 			foreach (i, inst; instructions){
80 				if (inst.args.length == 0 && inst.name[inst.name.length-1] == ':'){
81 					// make sure there's actually a name there, not just the colon
82 					if (inst.name.length <= 1){
83 						throw new Exception
84 							("line#"~to!string(i)~": jump position has no name");
85 					}
86 					r[cast(string)inst.name[0 .. inst.name.length-1].dup] = instructionCount;
87 					continue;
88 				}else{
89 					instructionCount ++;
90 				}
91 			}
92 			return r;
93 		}
94 		/// stores jump indexes
95 		uinteger[string] jumpIndexes;
96 		// first, convert it all to Function[]
97 		Function[] functionsByteCode = readByteCode(byteCode);
98 		/// makes the array to return
99 		auto functions = new LinkedList!(ByteCodeFunction!T);
100 		/// new instructions are appended to this
101 		auto instructions = new LinkedList!T;
102 		/// new instructions' args are appended to this
103 		auto instructionArgs = new LinkedList!(QData[]);
104 		// do the magic
105 		foreach (currentFunction; functionsByteCode){
106 			// get the jump positions
107 			try{
108 				jumpIndexes = searchJumpPos(currentFunction.instructions);
109 			}catch (Exception e){
110 				e.msg = "function: "~currentFunction.name~' '~e.msg;
111 				throw e;
112 			}
113 
114 			// convert deal with all the jumps
115 			foreach (i, inst; currentFunction.instructions){
116 				// check if it's a jump index
117 				if (inst.name == "jump"){
118 					// it's a jump 
119 					// the only arg is typeLess, and in string format, is name of the jumpIndex
120 					if (inst.args.length != 1){
121 						throw new Exception
122 							("function: "~currentFunction.name~" line#"~to!string(i)~": invalid arguments for jump instruction");
123 					}
124 					if (inst.args[0] !in jumpIndexes){
125 						throw new Exception
126 							("function: "~currentFunction.name~" line#"~to!string(i)~": invalid jump position");
127 					}
128 					inst.args[0] = "i"~to!string(jumpIndexes[inst.args[0]]-1);
129 				}
130 				// or if it's a jump position marker
131 				if (inst.name[inst.name.length-1] == ':'){
132 					continue;
133 				}
134 				// finally convert the instruction
135 				if (inst.name in instructionsPtr){
136 					instructions.append (instructionsPtr[inst.name]);
137 					instructionArgs.append (stringToQData(inst.args));
138 				}else{
139 					throw new Exception
140 						("function: "~currentFunction.name~" line#"~to!string(i)~": instruction '"~inst.name~"' not available");
141 				}
142 			}
143 			// put it all in a ByteCodeFunction
144 			functions.append(ByteCodeFunction!(void delegate())(
145 					currentFunction.name, // name
146 					instructions.toArray, // instructions
147 					instructionArgs.toArray // args
148 					));
149 			// clear stuff
150 			instructions.clear;
151 			instructionArgs.clear;
152 		}
153 		ByteCodeFunction!(T)[] r = functions.toArray;
154 		.destroy(functions);
155 		return r;
156 	}

161 /// reads "words" (instruction name, and arguments) into string[] from string
162 private static string[] readWords(string line){
163 	// first remove whitespace from it
164 	line = line.removeWhitespace('#');
165 	auto words = new LinkedList!string;
166 	for (uinteger i = 0, readFrom = 0, lastInd = line.length-1; i < line.length; i ++){
167 		// if a string, skip it
168 		if (line[i] == '"'){
169 			i = line.strEnd(i);
170 		}else if (line[i] == '['){
171 			i = line.bracketPos(i);
172 		}else if (line[i] == ' ' || line[i] == '\t'){
173 			words.append(line[readFrom .. i]);
174 			readFrom = i + 1;
175 			continue;
176 		}
177 		if (i >= lastInd){
178 			words.append (line[readFrom .. lastInd + 1]);
179 		}
180 	}
181 	string[] r = words.toArray;
182 	.destroy(words);
183 	return r;
