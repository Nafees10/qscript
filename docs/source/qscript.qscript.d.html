12 /// to store data from script at runtime
13 public union QData{
14 	string strVal; /// string value
15 	integer intVal; /// integer value
16 	double doubleVal; /// double/float value
17 	QData[] arrayVal; /// array value
18 	// constructor
19 	this (T)(T data){
20 		static if (is (T == string)){
21 			strVal = data;
22 		}else static if (is (T == int) || is (T == uint) || is (T == long) || is (T == ulong)){
23 			intVal = data;
24 		}else static if (is (T == double)){
25 			doubleVal = data;
26 		}else static if (is (T == QData[]) || is (T == void[])){
27 			arrayVal = cast(QData[])data;
28 		}else{
29 			throw new Exception("cannot store "~T.stringof~" in QData");
30 		}
31 	}
32 }

280 	void modifyArray(){
281 		QData[] indexes = currentCall.stack.pop!(true)(currentCall.readInstructionArgs[0].intVal);
282 		QData newVal, array;
283 		newVal = currentCall.stack.pop;
284 		array = currentCall.stack.pop;
285 		QData subArray;
286 		uinteger indexToModify = indexes[indexes.length-1].intVal;
287 		indexes.length--;
288 		array.arrayVal = array.arrayVal.dup;
289 		subArray = array;
290 		for (uinteger i=0; i < indexes.length; i ++){
291 			subArray = subArray.arrayVal[indexes[i].intVal];
292 		}
293 		subArray.arrayVal[indexToModify] = newVal;
294 		currentCall.stack.push(array);
295 	}

491 	/// to store data (args, result, vars) for a call to script-defined function, and the stack
492 	struct FunctionCallData{
493 		QData[] vars; /// the vars that this function has
494 		QData result; /// stores the return value of this function
495 		Stack!QData stack; /// the stack that this call will use
496 		
497 		string fName; /// name of this function
498 		uinteger instructionIndex; /// index of the current instruction being executed
499 		void delegate()[] instructions; /// the instructions that make this function
500 		QData[][] instructionArgs;
501 
502 		/// postblit
503 		this(this){
504 			vars = vars.dup;
505 			instructions = instructions.dup;
506 			instructionArgs = instructionArgs.dup;
507 		}
508 		/// constructor
509 		this(string functionName, QData[] args){
510 			vars = args.dup;
511 			fName = functionName;
512 			instructionIndex = 0;
513 			stack = new Stack!QData;
514 		}
515 		/// destructor
516 		~this(){
517 			.destroy(stack);
518 		}
519 		/// reads args for the current instruction
520 		@property QData[] readInstructionArgs(){
521 			return instructionArgs[instructionIndex];
522 		}
523 	}

541 	/// compiles and loads a script for execution.
542 	/// 
543 	/// Any previously loaded script/byte code is removed
544 	/// 
545 	/// `script` is the script to load, with each line as a separate element of the array, without the trailing newline character.
546 	/// 
547 	/// Returns: array of compile errors. or empty array in case of no errors
548 	CompileError[] loadScript(string[] script){
549 		// compile to byte code
550 		CompileError[] errors;
551 		string[] byteCode = compileScript(script, errors);
552 		if (errors.length > 0){
553 			return errors;
554 		}
555 		string sError;
556 		if (!loadByteCode(byteCode, sError)){
557 			return [CompileError(0, sError)];
558 		}
559 		return [];
560 	}

562 	/// compiles and loads a script for execution, while keeping the previously loaded script's or byte code's functions
563 	/// 
564 	/// `script` is the script to load, with each line as a separate element of the array, without the trailing newline character.
565 	/// `overWrite` if true, in case of functions with matching names, the ones from previously loaded one will be replaced with new one
566 	/// 
567 	/// Returns: array of compile errors. or empty array in case of no errors
568 	CompileError[] loadScript(bool overWrite)(string[] script){
569 		CompileError[] errors;
570 		// copy currently loaded script
571 		void delegate()[][string] loadedInstructions = scriptInstructions.dup;
572 		QData[][][string] loadedArgs = scriptInstructionsArgs.dup;
573 		// now just load it
574 		errors = loadScript(script);
575 		if (errors.length > 0){
576 			return errors;
577 		}
578 		// now load the saved script back accordingly to `overWrite`
579 		foreach (i, functionName; loadedInstructions.keys){
580 			// check if it exists in newly compiled one, case true: skip
581 			static if (overWrite){
582 				if (functionName in scriptInstructions){
583 					// skip
584 					continue;
585 				}
586 			}
587 			// load it back, no need to dup
588 			scriptInstructions[functionName] = loadedInstructions[functionName];
589 			// the args too
590 			scriptInstructionsArgs[functionName] = loadedArgs[functionName];
591 		}
592 	}

656 					"doubleToStr"		: &doubleToStr,
657 					"doubleToInt"		: &doubleToStr,

666 				]);
667 		}catch (Exception e){

669 			.destroy(e);
670 			return false;
671 		}
672 		// now put functions in the arrays
673 		foreach (currentFunction; functions){

680 	/// executes a script-defined function, with the provided arguments, and returns the result
681 	/// 
682 	/// throws Exception if the function does not exist
683 	QData executeScriptFunction(string fName, QData[] args){
684 		// check if it exists
685 		if (fName in scriptInstructions){
686 			// put previous call in call stack
687 			FunctionCallData* lastCall = currentCall;
688 			// prepare a new call
689 			currentCall = new FunctionCallData(fName, args);
690 			currentCall.instructions = scriptInstructions[fName];
691 			currentCall.instructionArgs = scriptInstructionsArgs[fName];
692 			// start executing instruction, one by one
693 			for (; currentCall.instructionIndex < currentCall.instructions.length; currentCall.instructionIndex ++){
694 				try{
695 					currentCall.instructions[currentCall.instructionIndex]();
696 				}catch (Exception e){
697 					if (!onRuntimeError(RuntimeError(fName, currentCall.instructionIndex, e.msg))){
698 						// break :(
699 						break;
700 					}
701 				}
702 			}
703 			// get the result of the function
704 			QData result = currentCall.result;
705 			// destroy stack
706 			.destroy(*currentCall);
707 			// restore previous call
708 			currentCall = lastCall;
709 			// return the result
710 			return result;
711 		}else{
712 			throw new Exception("function '"~fName~"' is not defined in the script/byteCode");
713 		}
714 	}
