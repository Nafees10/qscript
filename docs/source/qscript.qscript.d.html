12 /// to store data from script at runtime
13 public union QData{
14 	string strVal; /// string value
15 	integer intVal; /// integer value
16 	double doubleVal; /// double/float value
17 	QData[] arrayVal; /// array value
18 	// constructor
19 	this (T)(T data){
20 		static if (is (T == string)){
21 			strVal = data;
22 		}else static if (is (T == int) || is (T == uint) || is (T == long) || is (T == ulong)){
23 			intVal = data;
24 		}else static if (is (T == double)){
25 			doubleVal = data;
26 		}else static if (is (T == QData[]) || is (T == void[])){
27 			arrayVal = cast(QData[])data;
28 		}else{
29 			throw new Exception("cannot store "~T.stringof~" in QData");
30 		}
31 	}
32 }

485 	/// to store data (args, result, vars) for a call to script-defined function, and the stack
486 	struct FunctionCallData{
487 		QData[] vars; /// the vars that this function has
488 		QData result; /// stores the return value of this function
489 		Stack!QData stack; /// the stack that this call will use
490 		
491 		string fName; /// name of this function
492 		uinteger instructionIndex; /// index of the current instruction being executed
493 		void delegate()[] instructions; /// the instructions that make this function
494 		QData[][] instructionArgs;
495 
496 		/// postblit
497 		this(this){
498 			vars = vars.dup;
499 			instructions = instructions.dup;
500 			instructionArgs = instructionArgs.dup;
501 		}
502 		/// constructor
503 		this(string functionName, QData[] args){
504 			vars = args.dup;
505 			fName = functionName;
506 			instructionIndex = 0;
507 			stack = new Stack!QData;
508 		}
509 		/// destructor
510 		~this(){
511 			.destroy(stack);
512 		}
513 		/// reads args for the current instruction
514 		@property QData[] readInstructionArgs(){
515 			return instructionArgs[instructionIndex];
516 		}
517 	}

536 	CompileError[] loadScript(string[] script){
537 		// compile to byte code
538 		CompileError[] errors;
539 		string[] byteCode = compileScript(script, errors);
540 		if (errors.length > 0){
541 			return errors;

562 		ByteCode.ByteCodeFunction!(void delegate())[] functions;
563 		try{
564 			functions = ByteCode.toFunctionPtr!(void delegate())(byteCode,[
565 					// operators
566 					"addInt" 			: &addInt,
567 					"addDouble" 		: &addDouble,
568 					"subtractInt"		: &subtractInt,
569 					"subtractDouble"	: &subtractDouble,

572 					"divideInt"			: &divideInt,
573 					"divideDouble"		: &divideDouble,
574 					"modInt"			: &modInt,
575 					"modDouble"			: &modDouble,
576 					"concatArray"		: &concatArray,
577 					"concatString"		: &concatString,
578 					// bool operators
579 					"isSameInt"			: &isSameInt,
580 					"isSameDouble"		: &isSameDouble,
581 					"isSameString"		: &isSameString,
582 					"isLesserInt"		: &isLesserInt,
583 					"isLesserDouble"	: &isLesserDouble,
584 					"isGreaterInt"		: &isGreaterInt,
585 					"isGreaterDouble"	: &isGreaterDouble,
586 					"not"				: &not,

590 					"push"				: &push,

592 					"pop"				: &pop,
593 					"jump"				: &jump,
594 					"skipTrue"			: &skipTrue,
595 					"return"			: &returnInstruction,
596 					// arrays stuff
597 					"setLen"			: &setLen,
598 					"getLen"			: &getLen,
599 					"readElement"		: &readElement,
600 					"modifyArray"		: &modifyArray,
601 					"makeArray"			: &makeArray,
602 					"emptyArray"		: &emptyArray,
603 					// strings
604 					"strLen"			: &strLen,
605 					"readChar"			: &readChar,
606 					// concatString also comes under this, but it's an operator related thing, so its up there ^
607 					// data type conversion
608 					"strToInt"			: &strToInt,
609 					"strToDouble"		: &strToDouble,
610 					"intToStr"			: &intToStr,
611 					"intToDouble"		: &intToDouble,
612 					"doubleToStr"		: &doubleToStr,
613 					"doubleToInt"		: &doubleToStr,
614 					// executing functions
615 					"execFuncS"			: &execFuncS,
616 					"execFuncE"			: &execFuncE,
617 					// vars
618 					"varCount"			: &varCount,
619 					"getVar"			: &getVar,
620 					"setVar"			: &setVar
621 					
622 				]);
623 		}catch (Exception e){
624 			error = e.msg;
625 			.destroy(e);
626 			return false;
627 		}
628 		// now put functions in the arrays
629 		foreach (currentFunction; functions){
630 			scriptInstructions[currentFunction.name] = currentFunction.instructions.dup;
631 			scriptInstructionsArgs[currentFunction.name] = currentFunction.args.dup;
632 		}
633 		return true;
634 	}

636 	/// executes a script-defined function, with the provided arguments, and returns the result
637 	/// 
638 	/// throws Exception if the function does not exist
639 	QData executeScriptFunction(string fName, QData[] args){
640 		// check if it exists
641 		if (fName in scriptInstructions){
642 			// put previous call in call stack
643 			FunctionCallData* lastCall = currentCall;
644 			// prepare a new call
645 			currentCall = new FunctionCallData(fName, args);
646 			currentCall.instructions = scriptInstructions[fName];
647 			currentCall.instructionArgs = scriptInstructionsArgs[fName];
648 			// start executing instruction, one by one
649 			for (; currentCall.instructionIndex < currentCall.instructions.length; currentCall.instructionIndex ++){
650 				try{
651 					currentCall.instructions[currentCall.instructionIndex]();
652 				}catch (Exception e){
653 					if (!onRuntimeError(RuntimeError(fName, currentCall.instructionIndex, e.msg))){
654 						// break :(
655 						break;
656 					}
657 				}
658 			}
659 			// get the result of the function
660 			QData result = currentCall.result;
661 			// destroy stack
662 			.destroy(*currentCall);
663 			// restore previous call
664 			currentCall = lastCall;
665 			// return the result
666 			return result;
667 		}else{
668 			throw new Exception("function '"~fName~"' is not defined in the script/byteCode");
669 		}
670 	}
