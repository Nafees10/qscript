44 /// compiles a script from string[] to bytecode (in string[]).
45 /// 
46 /// `script` is the script to compile
47 /// `predefinedFunctions` is an array containing data about a function in Function[]
48 /// `errors` is the array to which errors will be appended
49 public string[] compileQScriptToByteCode(string[] script, Function[] predefinedFunctions, ref CompileError[] errors){
50 	/// called by ASTGen to get return type of pre-defined functions
51 	DataType onGetReturnType(string name, DataType[] argTypes){
52 		// some hardcoded stuff
53 		if (name == "setLength"){
54 			// return type will be same as the first argument type
55 			if (argTypes.length != 2){
56 				throw new Exception ("setLength called with invalid arguments");
57 			}
58 			return argTypes[0];
59 		}else if (name == "getLength"){
60 			// int
61 			return DataType(DataType.Type.Integer);
62 		}else if (name == "array"){
63 			// return array of the type received
64 			if (argTypes.length < 1){
65 				throw new Exception ("cannot make an empty array using 'array()', use [] instead");
66 			}
67 			DataType r = argTypes[0];
68 			r.arrayNestCount ++;
69 			return r;
70 		}else if (name == "strLen"){
71 			// returns length, int
72 			return DataType(DataType.Type.Integer);
73 		}else if (name == "return"){
74 			return DataType(DataType.Type.Void);
75 		}else{
76 			// search in `predefinedFunctions`
77 			foreach (currentFunction; predefinedFunctions){
78 				if (currentFunction.name == name){
79 					return currentFunction.returnType;
80 				}
81 			}
82 			throw new Exception ("function '"~name~"' not defined");
83 		}
84 	}
85 	/// called by ASTGen to check if a pre-defined function is called with arguments of ok-type
86 	bool onCheckArgsType(string name, DataType[] argTypes){
87 		// over here, the QScript pre-defined functions are hardcoded :(
88 		if (name == "setLength"){
89 			// first arg, must be array, second must be int
90 			if (argTypes.length == 2 && argTypes[0].arrayNestCount > 0 &&
91 				argTypes[1].type == DataType.Type.Integer && argTypes[1].arrayNestCount == 0){
92 				return true;
93 			}else{
94 				return false;
95 			}
96 		}else if (name == "getLength"){
97 			// first arg, the array, that's it!
98 			if (argTypes.length == 1 && argTypes[0].arrayNestCount > 0){
99 				return true;
100 			}else{
101 				return false;
102 			}
103 		}else if (name == "array"){
104 			// must be at least one arg
105 			if (argTypes.length < 1){
106 				return false;
107 			}
108 			// and all args must be of same type
109 			foreach (type; argTypes){
110 				if (type != argTypes[0]){
111 					return false;
112 				}
113 			}
114 			return true;
115 		}else if (name == "strLen"){
116 			// one arg, must be string
117 			if (argTypes.length != 1){
118 				return false;
119 			}else if (argTypes[0] == DataType(DataType.Type.String)){
120 				return true;
121 			}
122 			return false;
123 		}else if (name == "return"){
124 			// real type checking for this was done at compile-time (AST generator)
125 			if (argTypes.length == 1){
126 				return true;
127 			}
128 			return false;
129 		}else{
130 			// now check the other functions
131 			foreach (currentFunction; predefinedFunctions){
132 				if (currentFunction.name == name){
133 					return matchArguments(currentFunction.argTypes.dup, argTypes);
134 				}
135 			}
136 			throw new Exception ("function '"~name~"' not defined");
137 		}
138 	}
139 	// add some other predefined QScript functions
140 	predefinedFunctions = predefinedFunctions.dup;
141 	predefinedFunctions ~= [
142 		Function("strToInt", DataType(DataType.Type.Integer), [DataType(DataType.Type.String)]),
143 		Function("strToDouble", DataType(DataType.Type.Double), [DataType(DataType.Type.String)]),
144 		
145 		Function("intToStr", DataType(DataType.Type.String), [DataType(DataType.Type.Integer)]),
146 		Function("intToDouble", DataType(DataType.Type.Double), [DataType(DataType.Type.Integer)]),
147 		
148 		Function("doubleToStr", DataType(DataType.Type.String), [DataType(DataType.Type.Double)]),
149 		Function("doubleToInt", DataType(DataType.Type.Integer), [DataType(DataType.Type.Double)])
150 	];
151 	// first convert to tokens
152 	errors = [];
153 	TokenList tokens = toTokens(script.dup, errors);
154 	if (errors.length > 0){
155 		return [];
156 	}
157 	// then generate the AST
158 	ScriptNode scriptAST;
159 	ASTGen astGenerator = ASTGen(&onGetReturnType, &onCheckArgsType);
160 	// generate the ast
161 	scriptAST = astGenerator.generateScriptAST(tokens, errors);
162 	if (errors.length > 0){
163 		return [];
164 	}
165 	// now finally the byte code
166 	CodeGen byteCodeGenerator;
167 	string[] code = byteCodeGenerator.generateByteCode(scriptAST, errors);
168 	if (errors.length > 0){
169 		return [];
170 	}
171 	return code;
172 }
